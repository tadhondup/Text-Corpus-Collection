མཆོད་པ་བརྗོད་དང་གླེང་གཞི་བསྟན་པ།
 
རབ་འབྱམས་ཕྱོགས་བཅུའི་རྒྱལ་བ་ཐམས་ཅད་ཀྱི།
མཁྱེན་བརྩེའི་ཡེ་ཤེས་ཕུང་པོ་བསྡུས་པ་བཞིན།
སྙིགས་དུས་བསྟན་དང་འགྲོ་བའི་དཔལ་གཅིག་པུ།
ཀུན་མཁྱེན་རྩ་བའི་བླ་མས་བདེ་ལེགས་སྩོལ།
 
        ནོར་འཛིན་རིན་ཆེན་དང་པོའི་ཁྲི་ལ་རཏྞ་པཉྩའི་འདབ་མ་འབུམ་གཡོའི་གེ་སར་རྩེར། དཀར་དམར་ཉི་ཟླའི་གདན་ལ་མཛེས་བརྗིད་གང་སྐུ་ས་ལེ་སྦྲམ་གྱི་མདོག་ལྟར་རབ་གསལ་བ། བླ་མེད་གཙུག་གཏོར་མངོན་པར་མཐོ་སོགས་མ་བྲིས་མཚན་མཆོག་བཟང་པོ་སོ་གཉིས་རྣམ་པར་བཀྲ། ཡན་ལག་དྲུག་ཅུའི་གསུང་དབྱངས་རབ་འཇེབས་བརྒྱ་ཡི་མགོན་དེས་རྣམ་པ་ཀུན་ཏུ་དགེ་ལེགས་སྩོལ། ངེས་དོན་མཐོང་བའི་འདྲེན་བྱེད་དང་བྲལ་འཁོར་བའི་སྨག་རུམ་འཐོམས་པའི་ཉམ་ཐག་འགྲོར། སྙིང་རྗེ་ཆེན་པོས་ཡལ་བར་མི་འདོར་ཚད་མེད་ཡེ་ཤེས་སྤྱན་གྱིས་ཉེར་གཟིགས་ནས། ཐབས་དང་ཐུགས་རྗེའི་འོད་སྟོང་སྤྲོས་ནས་ཐར་བའི་ཞིང་དུ་བཀྲིད་ལ་རབ་མཁས་པའི། བརྟན་གཡོ་དབང་སྒྱུར་ཀུན་བཟང་དུས་ཀྱི་འཁོར་ལོར་སྒོ་གསུམ་གུས་པས་ཕྱག་བགྱིས་སོ།། མར་ཞི་ལས་གྲུབ་གཡུ་ཡི་ཕྲ་བཀོད་རྩེ་ཕྲན་རྣམ་ལྔའི་ཐོད་མཛེས་ཤིང་། གང་སྐུ་གཡས་སུ་འགྱིང་བའི་ཟོལ་གྱིས་ཤེས་རབ་རལ་ཀྱུ་མཁའ་ལ་ཕྱར། གཡོན་པས་ཀེ་ས་ར་དབུས་བླ་མེད་ཤེར་ཕྱིན་པོ་ཏིས་རབ་མཚོན་ཞིང་། པཉྩ་ལི་ཀའི་རྒྱན་གྱིས་རབ་ཀླུབས་སྨྲ་བའི་ལྷ་ཁྱོད་རྒྱལ་གྱུར་ཅིག། རབ་འཇེབས་ཏམ་བུ་ར་ཡི་དབྱངས་ཀྱིས་ཐོས་འཛིན་སྙན་པའི་དཔལ་སྟེར་ཞིང་། བཅུ་དྲུག་ལང་ཚོའི་ཉམས་ཀྱིས་འགྲོ་བའི་ཡིད་འཕྲོག་སྒྲ་དབྱངས་ལྷ་མོ་ཞེས། མཚན་ཙམ་ཐོས་པས་སྔོན་མེད་དད་པའི་སྤུ་ལོང་རབ་ཏུ་རྒྱས་པར་བྱེད། བཅུ་ཕྲག་རིག་གནས་སྨྲ་བའི་ཚངས་དབྱངས་སྟེར་མཛད་དེ་ཉིད་མགྲིན་པར་རོལ། ཕན་བདེའི་འབྱུང་གནས་རྒྱལ་བསྟན་ཟབ་ལམ་ལྷ་ཡི་རྫིང་རིང་པོར། ཐོས་བརྩོན་ནམ་མཁའི་གོས་ལས་ལེགས་བཤད་སྤྲིན་ཟེགས་མངོན་འབབ་པས། ལུས་ཅན་འཛིན་མའི་ཁྱོད་ལས་བྱང་ཆུབ་ཆུ་སྐྱེས་བཞད་མཛད་པ། རྟག་གཟིགས་ཞང་བོད་མཁས་མཆོག་ཀླུ་དབང་ཚོགས་ཀུན་རྒྱལ་གྱུར་ཅིག། ངེས་ངེས་ཆེ་བའི་རིགས་རུས་ཡོངས་འདུས་ལྗོན་པ་ཆེར། ངེས་ངེས་སྦྱང་རྟོགས་ཡོན་ཏན་ལོ་འབྲས་གཡུར་ཟ་ཞིང་། མཁས་མཁས་སྙན་པའི་དྲི་བསུང་ཕྱོགས་བརྒྱར་རབ་འཐུལ་བའི། མཁས་མཁས་བྲུ་ཞུ་སྤ་རྨེའུ་གཤེན་ལྔ་སྲིད་འདིའི་རྒྱན། སྐུ་གསུམ་བདེ་ཆེན་ཡེ་ཤེས་མཁའ་ལ་དུས་གསུམ་བདེ་གཤེགས་མཁྱེན་པའི་འོད་སྟོང་གཅིག་བསྡུས་ནས། གསང་གསུམ་མི་བཟད་དཀྱིལ་འཁོར་རབ་རྒྱས་ཚད་གསུམ་ལུང་རྟོགས་བོན་གྱི་ཚ་ཟེར་འབུམ་འགྱེད་པས། སྒྲིབ་གསུམ་ལས་བསྐྱེད་དུཿ ཁས་མནར་བའི་ཁམས་གསུམ་མུན་པའི་ར་བར་འཐོམས་ཀུན་རབ་གསལ་ཞིང་། མཆོག་གསུམ་རྒྱལ་བའི་ཞིང་དུ་འགོད་མཛད་སྲིད་གསུམ་རྒྱན་གྱུར་མཉམ་མེད་མཚན་ཅན་ལན་བརྒྱར་བསྟོད། མཁས་ཀུན་གཙུག་རྒྱན་རྒྱལ་བ་གཉིས་པའི་རིང་ལུགས་ཉམས་མེད་རྣམ་རྒྱལ་ཁང་བཟང་རྩེར། མཁས་པའི་བྱ་བ་གསུམ་གྱི་བ་དན་དཀར་པོ་སྲིད་རྩེར་རབ་ཏུ་བསྒྲེངས་བྱེད་པའི། མཁས་མཆོག་གཟུགས་མིག་གྲངས་ལྡན་སྤྱི་དང་ཁྱད་པར་བདག་བློའི་ཀུན་མུད་བཞད་མཛད་པ། མཁས་དབང་པུཎྜེ་མ་ཏིའི་མཚན་སོགས་དྲིན་ཅན་ཡོངས་འཛིན་ཚོགས་དེར་དད་ཕྱག་རྩེགས། ལྷབ་ལྷབ་མེ་རླུང་འཚུབ་མའི་དཀྱིལ་ནས་འུར་འུར་འཇིགས་པའི་སྒྲ་སྒྲོགས་པས། ཅུམ་ཅུམ་བདུད་སྡེ་དཔུང་བཅས་འདར་ནས་ཆིལ་ཆིལ་སྙིང་ཁྲག་འབྱིན་བྱེད་པའི། ལྷང་ལྷང་བསྟན་སྲུང་རྒྱ་མཚོའི་ཚོགས་ལ་དུང་དུང་གདུང་བས་གུས་བསྟོད་པས། མཛེས་མཛེས་འཛུམ་ཞལ་གཤེན་ལ་སྟོན་ནས་ཡང་ཡང་སྡོང་དང་གྲོགས་མཛོད་ཅིག། དེ་ལྟར་བསྟོད་ཚིག་ཁྱུང་གི་ཕུ་བོ་ཡིས། ཉེ་བར་དྲངས་ཏེ་གཙུག་ལག་སྐར་རྩིས་ཀྱི། ལག་ལེན་སྣང་སྲིད་གསལ་བའི་སྒྲོན་མེ་ཞེས། ངོ་མཚར་འོད་ཟེར་འབུམ་ལྡན་འདི་ན་སྤྲོ། དེ་ཡང་རིག་གནས་ཆུང་བ་ལྔའི། ནང་ཚན་སྐར་རྩིས་རིག་པ་ཡི། འོས་པའི་གླེང་གཞིའི་རིམ་པ་ནི། ཐོག་མར་བོན་དབྱིངས་འདུས་མ་བྱས། སྤྲོས་བྲལ་ནམ་མཁའ་ལྟ་བུ་ལས། རྫོགས་པའི་སྐུར་བཞེངས་ཀུན་སྣང་ཁྱབ། བདེར་གཤེགས་དེ་ཡི་ཐུགས་རྗེ་ལས། འབྱུང་ལྔ་སྣོད་བཅུད་རིམ་ཆགས་པའི། ད་ལྟའི་སྐལ་པ་བཟང་པོ་འདིར། ཐར་ལམ་བདེ་འབྱེད་རྫོགས་སངས་རྒྱས། སྟོང་དང་རྩ་གཉིས་བྱོན་པ་ལས། ཁྱད་པར་སྙིགས་མ་ལྔ་བདོ་བའི། ལུས་ཅན་བརྩེ་བས་མི་འདོར་ཕྱིར། སྟོན་མཆོག་བརྒྱ་བའི་མགོན་པོ་དེས། དང་པོར་བྱང་ཆུབ་ཐུགས་བསྐྱེད་དེ། གྲངས་མེད་གསུམ་དུ་ཚོགས་གཉིས་བསགས། མཐར་ཐུག་མངོན་པར་སངས་རྒྱས་ནས། བོན་གྱི་སྐུ་ལས་མ་གཡོས་ཀྱང་། ལོངས་སྤྱོད་རྫོགས་པའི་སྐུར་བཞེངས་ཏེ། རིགས་ལྔ་བདེར་གཤེགས་ཞི་དང་ཁྲོ། བརྗོད་འདས་རྒྱལ་བའི་དཀྱིལ་འཁོར་སྤྲོས། སླར་ཡང་འགྲོ་བ་རང་མཐུན་དུ། སྤྲུལ་ནས་སེམས་ཅན་དོན་མཛད་ཕྱིར། ཚད་མེད་སྙིང་རྗེ་ཆེན་པོ་ཡིས། རྫོགས་སྐུ་གཤེན་ལྷའི་ལུང་ནོད་ཅིང་། སྲིད་པ་སངས་པོ་འབུམ་ཁྲི་དང་། ཐབས་དང་ཤེས་རབ་ཟུང་དུ་སྦྲེལ། གཙང་རིགས་ལྔ་ནས་རིམ་གཤེགས་ཏེ། རྣམ་པར་རྒྱལ་བའི་ཁང་བཟང་དུ། རྩ་གསུམ་ལྷ་ལ་བོན་འཁོར་བསྐོར། ལྷོ་ནུབ་རྩུབ་འགྱུར་དྲེགས་ཚལ་དུ། སྡེ་བརྒྱད་དྲེགས་ཚོགས་དམ་ལ་བཏགས། སངས་རྒྱས་བསྟན་པ་སྐྱོང་བར་བསྐོས། རི་བདུན་རྩེ་མོར་འཕགས་སྐྱེས་པོའི། ཕོ་བྲང་ཛཱ་རན་དྷ་རིའི་རྩེར། གཟིགས་ཚད་ལྔ་དང་དགོངས་ཚད་ལྔའི། རྟག་གཟིགས་འོལ་མོ་གླིང་ཆེན་དུ། ལོ་རྒྱལ་ཤིང་བྱི་དཔྱིད་རའི་ཉར། བྱམས་མའི་རྣམ་སྤྲུལ་རྒྱལ་བཞད་མའི། སྲས་སུ་མངོན་པར་སྐུ་བལྟམས་ནས། རྒྱལ་བའི་མཛད་ཆེན་བཅུ་གཉིས་དང་། མཛད་ཕྲན་དྲུག་ཅུའམ་བརྒྱ་རྩ་བརྒྱད། བོན་སྒོ་བརྒྱད་ཁྲི་བཞི་སྟོང་དང་། སྤྲུལ་སྐུ་བྱེ་བ་ཕྲག་བརྒྱ་ཡི། སེམས་ཅན་འགྲོ་དོན་རྒྱ་ཆེར་མཛད། ཁྱད་པར་གཤེན་ལོ་བཅུ་གཉིས་ཏེ། ས་ཡི་ལྷ་མོ་བརྟན་མ་ཡིས། གསེར་བུམ་བདུད་རྩིས་བལྟམས་པ་ཞིག། བཀྲ་ཤིས་རྫས་སུ་མངའ་དབུལ་ནས། རྟག་གཟིགས་འོལ་མོ་གླིང་ཆེན་གྱི། ལྷོ་ཕྱོགས་མོན་ཡུལ་སུམ་སྟོད་དུ། དབང་ཆེན་ས་བདག་རྒྱལ་པོ་ཡིས། ཕོ་བྲང་ཆེན་པོར་སྤྱན་དྲངས་ཚེ། གཙང་ཀུན་ཁྱབ་པ་དབང་ཆེན་གྱིས། གསེར་གྱི་འཁོར་ལོ་རྩིབས་སྟོང་ལ། ཉི་ཟླ་ནོར་བུས་བརྒྱན་པ་ཕུལ། བསྟན་པ་སྤེལ་བར་གསོལ་བ་བཏབ། དེ་ཚེ་སྟོན་མཆོག་གཤེན་རབ་ཀྱིས། དུས་དང་རྟེན་འབྲེལ་འཛོམས་པ་དང་། བསྟན་པ་རབ་ཏུ་བསྟན་བྱའི་ཕྱིར། དབང་ལྡན་དུས་ཀྱི་འཁོར་ལོ་ཡི། གཞལ་ཡས་བཞེངས་པར་བཀའ་ཡང་བསྩལ། སྡོང་ཟླ་ལྷ་ཀླུ་མིས་བྱས་ཏེ། བཟོ་རྒྱལ་གར་མ་ལི་ཤོ་དང་། འཛམ་གླིང་མཛེས་བྱེད་རྒྱན་དྲུག་གིས། ལེགས་པར་གཟབས་ཏེ་བཞེངས་པ་ཡིས། ལྕགས་རིར་གྲུ་བཞིའི་ལོགས་རེ་ལ། གྱད་འདོམས་བརྒྱ་དང་བཞི་བཅུ་རེ། བསྐོར་བའི་དཀྱིལ་དུ་ཕོ་བྲང་ནི། གསས་མཁར་བ་རབ་ཡོངས་འདུས་ཞེས། ལྷ་དང་ལྷ་མིན་ཀླུ་དང་སྲིན། སྲིད་བསྐོས་ཕྱ་ལ་དཔེ་བླང་བའི། སྣ་ཚོགས་བཟོ་རིགས་ཀུན་འདུས་ལ། དབུ་མ་རྒྱལ་བའི་དཀྱིལ་འཁོར་ཏེ། དུས་ཀྱི་འཁོར་ལོའི་ཞིང་ཁམས་དང་། བར་འཁྱམས་འདོད་ཡོན་ར་བ་ལ། མཆོད་པའི་བྱེ་བྲག་སྣ་ཚོགས་ལྡན། ཕྱི་འཁྱམས་སྤར་ཁམས་སྨེ་གླིང་དང་། ལོ་སྒོར་བཅུ་གཉིས་ལ་སོགས་ཏེ། དབང་ཆེན་ས་བདག་སྡེ་ཚོགས་དང་། དཔལ་མགོན་བདུད་ཅུ་རྒྱལ་ཆེན་བཞིའི། ཕོ་བྲང་ཞིང་གི་བཀོད་པ་སོགས། ངོ་མཚར་ཡིད་འོང་རྒྱ་ཆེའི་གནས། རྩ་གསུམ་ལྷ་ལས་དཔག་པ་དེར། ཆུ་ཕག་ནག་ཟླའི་ཉ་ཚེས་ལ། སྟོན་མཆོག་བརྒྱ་བའི་མགོན་པོ་དེས། ཞུ་འཁོར་ཚངས་པ་གཙུག་ཕུད་དང་། གཙུག་ལག་ཞུ་བའི་གཤེན་བརྒྱད་དང་། ཉན་འཁོར་གཡུང་དྲུང་སེམས་དཔའ་སྟེ། ཡེ་གཤེན་སུམ་བརྒྱ་དྲུག་ཅུ་དང་། སེམས་དཔའ་ལྔ་སྟོང་ལྔ་བརྒྱ་དང་། གཤེན་ཕྲན་ཁྲི་དང་དྲུག་སྟོང་དང་། ལྷ་ཡི་དབང་པོ་བརྒྱ་བྱིན་དང་། ཀླུ་རྒྱལ་གཙུག་ན་རིན་ཆེན་དང་། མི་དབང་འདོད་པ་དགུ་འགྱུར་དང་། ལྷ་མིན་ཐང་བཟང་རིང་སྐྱེས་ཏེ། འཇིག་རྟེན་མངའ་དབང་ཆེ་བཞི་དང་། ས་བདག་ཀླུ་གཉན་གཏོད་བཞི་ཡིས། རྗེ་བློན་བཙུན་མོ་འབངས་འཁོར་བཅས། འཁོར་ཚོགས་བྱེ་བ་གྲངས་མེད་ལ། མཐོ་ཐོག་སྤྱི་རྒྱུག་མཛོད་ཀྱི་བོན། དབང་ལྡན་དུས་ཀྱི་འཁོར་ལོ་གསུངས། དེ་ཡང་མདོ་རྒྱུད་གཉིས་ཡིན་ཏེ། མདོ་ནི་བསྐལ་སྲིད་འབྲོས་འདེད་མདོ། རྒྱུད་ལ་སྒྲུབ་མཆོད་མན་ངག་གསུམ། སྒྲུབ་པ་དབང་དྲག་རྒྱས་པ་སྟེ། འཁོར་ལོ་རོལ་པ་སྐུ་ཡི་རྒྱུད། འཁོར་ལོ་དབང་ཕྱུག་གསུང་གི་རྒྱུད། འཁོར་ལོ་གསང་བ་ཐུགས་ཀྱི་རྒྱུད། རྒྱུད་སྲུང་ཀིང་དང་མགོན་པོ་དང་། དཔལ་འབར་ཨེ་ཀ་ཙཱ་ཏི་དང་། འཇིག་རྟེན་མངའ་དབང་བདག་པོས་སྐྱོངས། ལེའུ་སྟོང་ཕྲག་བརྒྱ་དང་བཅས། མཆོད་སྦྱིན་དུས་ཀྱི་འཁོར་ལོ་ནི། བདུན་བཅུ་རྩ་བཞི་རྩ་བའི་ལྷ། རྒྱལ་ཆེན་ཕྱོགས་ཀུན་སྐྱོངས་བཅས་པའི། སངས་རྒྱས་བསྟན་པ་བརྟན་པའི་ཕྱིར། རབ་ཏུ་གནས་པའི་ཆོ་ག་དང་། ཚོགས་བསགས་མཆོད་འབུལ་རིམ་པའོ།། མན་ངག་དུས་ཀྱི་འཁོར་ལོ་ནི། སྐལ་སྲིད་ཆགས་གནས་འཇིག་སྟོངས་དང་། དུས་བཞི་འབྱུང་བའི་དར་རྒུད་དང་། སྐྱེ་འཕེལ་ཚེ་རབས་ལས་རྩིས་དང་། སྔོན་དང་མ་འོངས་ད་ལྔ་དང་། ལོ་ཟླ་ཞག་གྲངས་དུས་ཚོད་ཀྱིས། ཀུན་རྫོབ་མངོན་པར་ཤེས་པ་སྟེ། གཙུག་ལག་རྩིས་ཀྱི་སྐོར་དང་བཅས། ས་བོན་གསེར་ཐུབ་འོད་སྲུང་ལ། བསྟན་པ་རིན་ཆེན་གཉེར་གཏད་དེ། རྟག་གཟིགས་ཤམ་བྷ་ལའི་ཞིང་དེར། ཤིན་ཏུ་དར་ཞིང་རྒྱས་པའོ།། དེ་ནས་གདུང་སོབ་ཆེན་པོ་ཡིས། ལོ་ཙ་མཁས་པ་རྒྱན་དྲུག་ལ། དགོངས་འགྲེལ་ཆ་ལག་དང་བཅས་ཏེ། ལེགས་པར་གཏན་ལ་ཕབ་ཏེ་བཤད། ལོ་ཙ་ཆེན་པོ་དེ་དྲུག་གིས། ཞང་ཞུང་རྒྱ་གར་རྒྱ་ནག་དུ། བསྒྱུར་ཞིང་བསྟན་པ་སྤེལ་བར་མཛད། བར་དེར་ཙི་ལུ་པཎ་ཆེན་གྱིས། ཆོས་ཀྱི་ལུགས་སུ་བསྒྱུར་བྱས་ཤིང་། བོན་བསྟན་དར་ནུབ་བྱུང་གྱུར་ཡང་། སྨོན་ལམ་རྣམ་དག་འབྲས་བུ་དང་། སེམས་ཅན་སྤྱི་མཐུན་བསོད་ནམས་ཀྱིས། བོན་བསྟན་ཕྱི་དར་བྱུང་བའི་དུས། མཆོད་སྦྱིན་དུས་ཀྱི་འཁོར་ལོ་དང་། གཙུག་ལག་དཔལ་འབུམ་སྐོར་དང་ནི། གཡེན་ཁམས་སེལ་བའི་མདོ་སྡེ་དང་། གཞན་ཡང་གཙུག་ལག་རྩིས་སྐོར་ལའང། གཏེར་དང་སྙན་རྒྱུད་ལས་ཐོན་ཞིང་། ཁྱད་པར་མཁན་ཆེན་བསོད་བློ་ལ། དག་སྣང་ལུང་བསྟན་རྩིས་ཀྱིས་འབུམ། ཤེལ་དཀར་ཟ་མ་ཏོག་ཞེས་པ། ཡོད་མོད་རྒྱས་པར་མ་ཕེབ་རུང་། མདོར་བསྡུས་འབྱུང་བཞིན་དེ་དག་དང་། གཤེན་བསྟན་སྨན་རིའི་ཕྱག་རྒྱུན་གྱིས། དགོང་དོན་ལག་ལེན་ཞལ་ཁྲིད་རྣམས། བུམ་པ་གང་བྱོའི་ཚུལ་ལྟ་བུར། རྫོགས་པར་བདག་ལ་བསྩལ་བར་བརྟེན། རང་དང་རང་འདྲའི་འགྲོ་རྣམས་ལ། ཕན་སེམས་སྐྱ་རེངས་ཀྱིས་དྲངས་ཏེ། ལག་ལེན་གསལ་སྒྲོན་འདི་བརྩམས་སོ།། ཞེས་སྐར་རྩིས་ཀྱི་ལེགས་པར་བཤད་པ་སྣང་སྲིད་གསལ་བའི་སྒྲོན་མ་ངོ་མཚར་འོད་ཟེར་འབུམ་ལྡན་ལས། གླེང་གཞིའི་ལོ་རྒྱུས་མདོ་ཙམ་བསྟན་པའི་སྐབས་ཏེ་དང་པོའོ།།
        ཉེར་མཁོ་སྔོན་འགྲོའི་རིམ་པ་མདོ་ཙམ་བཤད་པ།
        དེ་ནས་སྔོན་འགྲོའི་རིམ་པ་ལ། ཐོག་མར་དགུ་མཐའི་ངེས་པ་ནི། བསྒྱུར་བཀོད་རི་མོའི་ལག་ལེན་རྣམས། རང་བྱན་ཆུད་པར་བྱ་དགོས་ཕྱིར། དགུ་གཅིག་གཅིག་དགུ་དགུ་རང་ལ། དགུ་གཉིས་གཉིས་དགུ་བཅོ་བརྒྱད་ཡིན། དགུ་གསུམ་གསུམ་དགུ་ཉི་ཤུ་བདུན། དགུ་བཞི་བཞི་དགུ་སུམ་ཅུ་དྲུག། དགུ་ལྔ་ལྔ་དགུ་བཞི་བཅུ་ལྔ། དགུ་དྲུག་དྲུག་དགུ་ལྔ་བཅུ་བཞི། དགུ་བདུན་བདུན་དགུ་དྲུག་ཅུ་གསུམ། དགུ་བརྒྱད་བརྒྱད་དགུ་བདུན་ཅུ་གཉིས། དགུ་དགུ་བརྒྱད་ཅུ་གྱ་གཅིག་གོ། བརྒྱད་གཅིག་གཅིག་བརྒྱད་བརྒྱད་རང་ཡིན། བརྒྱད་གཉིས་གཉིས་བརྒྱད་བཅུ་དྲུག་ལ། བརྒྱད་གསུམ་གདུམ་བརྒྱད་ཉི་ཤུ་བཞི། བརྒྱད་བཞི་བཞི་བརྒྱད་སུམ་ཅུ་གཉིས། བརྒྱད་ལྔ་ལྔ་བརྒྱད་བཞི་བཅུ་ཐམ། བརྒྱད་དྲུག་དྲུག་བརྒྱད་བཞི་བཅུ་བརྒྱད། བརྒྱད་བདུན་བདུན་བརྒྱད་ལྔ་བཅུ་དྲུག། བརྒྱད་བརྒྱད་དྲུག་ཅུ་རྩ་བཞིའོ།། བདུན་གཅིག་གཅིག་བདུན་བདུན་རང་ཡིན། བདུན་གཉིས་གཉིས་བདུན་བཅུ་བཞི་ལ། བདུན་གསུམ་གསུམ་བདུན་ཉི་ཤུ་གཅིག། བདུན་བཞི་བཞི་བདུན་ཉི་ཤུ་བརྒྱད། བདུན་ལྔ་ལྔ་བདུན་སུམ་ཅུ་ལྔ། བདུན་དྲུག་དྲུག་བདུན་བཞི་བཅུ་གཉིས། བདུན་བདུན་བཞི་བཅུ་ཞེ་དགུའོ།། དྲུག་གཅིག་གཅིག་དྲུག་དྲུག་རང་ཡིན། དྲུག་གཉིས་གཉིས་དྲུག་བཅུ་གཉིས་ཏེ། དྲུག་གསུམ་གསུམ་དྲུག་བཅོ་བརྒྱད་ལ། དྲུག་བཞི་བཞི་དྲུག་ཉི་ཤུ་བཞི། དྲུག་ལྔ་ལྔ་དྲུག་སུམ་ཅུ་ཡིན། དྲུག་དྲུག་སུམ་ཅུ་སོ་དྲུག་གོ། ལྔ་གཅིག་གཅིག་ལྔ་ལྔ་ཉིད་དེ། ལྔ་གཉིས་གཉིས་ལྔ་བཅུ་ཐམ་པ། ལྔ་གསུམ་གསུམ་ལྔ་བཅོ་ལྔ་ལ། ལྔ་བཞི་བཞི་ལྔ་ཉི་ཤུ་ཐམ། ལྔ་ལྔ་ཉི་ཤུ་རྩ་ལྔའོ།། བཞི་གཅིག་གཅིག་བཞི་བཞི་རང་ཡིན། བཞི་གཉིས་གཉིས་བཞི་བརྒྱད་ཡིན་ལ། བཞི་གསུམ་གསུམ་བཞི་བཅུ་གཉིས་སོ།། བཞི་བཞི་བཅུ་དྲུག་དག་དུ་འདོད། གསུམ་གཅིག་གཅིག་གསུམ་གསུམ་ཉིད་དེ། གསུམ་གཉིས་གཉིས་གསུམ་དྲུག་ཡིན་ལ། གསུམ་གསུམ་དགུ་རུ་གྲགས་པ་ཉིད། གཉིས་གཅིག་གཅིག་གཉིས་གཉིས་ཡིན་ལ། གཉིས་གཉིས་དག་ལ་བཞི་ཞེས་གྲགས། གཅིག་གཅིག་གཅིག་ནི་གཅིག་ཉིད་དོ།། འབྲི་བགྲང་སྣོན་པ་གཡས་ནས་བརྩམས། བསྒྱུར་བགོད་འབྲིད་པ་གཡོན་ནས་བྱ། སྒྱུར་བྱེད་ཐིག་ལེ་གྲལ་མཐར་བཀོད། བགོད་བྱེད་ཐིག་ལེས་མས་མཐའ་བཀག། བསྒྱུར་བྱ་སྟོང་ན་བསྒྱུར་རྒྱུ་མེད། སྒྱུར་བྱེད་སྟོང་ན་བསྒྱུར་གཞི་དབྱི། བསྒྱུར་བྱའི་འོག་ཏུ་སྒྱུར་བྱེད་ཀྱི། ཐ་མ་བཞག་ནས་རིམ་བཞིན་བསྒྱུར། གཅིག་གིས་བསྒྱུར་ལ་འཕེལ་འགྲིབ་མེད། གཉིས་ལ་སོགས་པས་བསྒྱུར་བ་ན། སྒྱུར་བྱེད་གྲངས་སུ་བསྒྱུར་བྱ་འཕེལ། བགོད་བྱེད་མཐའ་དབུས་ཐིག་ལེས་ནི། རང་གནས་འཛིན་ལས་བགོད་ནུས་མེད། མ་ཐོབ་ནོར་སྐལ་ཐིག་ལེས་བྱིན། བགོད་ལ་རྐྱང་མ་སྦྲགས་མ་གཉིས། རྐྱང་བགོད་ཇི་ཙམ་ཐོབ་པ་ཁྱེར། སྦྲགས་བགོད་ལ་ནི་འགོ་མ་གཙོ། བགོད་པའི་དོན་ནི་བགོ་བྱ་སྟེ། བགོད་བྱེད་གྲངས་སུ་སྡུད་པ་ཡིན། ཐི་ལེ་གྲལ་འགོར་འབྱུང་མི་སྲིད། གཟའ་དང་བྱེད་པའི་འཁོར་ལོ་བདུན། སྐར་སྦྱོར་གཉིས་ནི་ཉེར་བདུན་ནོ།། ཆུ་ཚོད་ཆུ་སྲང་དྲུག་ཅུ་ལ། དབུགས་དྲུག་ཆ་ཤས་མང་ཉུང་དང་། གནས་ལ་ངེས་པ་མེད་པ་ཡིན། བསྒྱུར་དང་བསྣུན་དང་སྤེལ་བ་དང་། བསྒྲེས་དང་གསིལ་ལྔ་དོན་གཅིག་ལ། བགོ་དང་ཕྱེ་དང་བསལ་བ་དང་། བཅད་པ་འདི་བཞི་དོན་གཅིག་ཡིན། བསྲེ་བ་སྟེར་བ་ཕུལ་བ་དང་། བྱིན་དང་བརྒྱན་དང་ཁན་པ་དང་། བསྣན་པ་སྣོན་པ་ནོར་དུ་འགྱུར། ལྡན་པ་འདི་བཅུ་དོན་གཅིག་གོ། ཕྲི་དང་སྦྱང་དང་འཕྲོག་པ་དང་། ཞུ་དང་ཕྲལ་དང་དོར་བ་དང་། འབྲིད་པ་དམན་པ་བུ་ལོན་འགྱུར། དབེན་པ་འདི་བཅུ་དོན་གཅིག་ཡིན། ནོར་དེ་དོར་བའི་གོ་ཤུགས་ཀྱིས། བགོད་ལ་དོར་བ་འཇུག་པའང་ཡོད། རྙེད་པ་ཐོབ་པ་ནོར་དང་གསུམ། དོན་གཅིག་ཏུ་ནི་ཤེས་པ་བྱ། དེ་ནས་གྲངས་ཀྱིས་བརྡའ་ཆད་ལ། མི་མཐུན་རྣམ་གྲངས་མང་ན་ཡང་། ཀུན་མཁྱེན་མཉམ་མེད་བླ་མ་ཡི། མཛོད་འགྲེལ་འཕྲུལ་གྱི་སྒྲོན་མེ་ལས། གྲངས་མིག་དྲུག་ཅུ་ཐམ་པ་རུ། གསུངས་པའི་དགོངས་དོན་འདི་ལྟར་ཏེ། གཅིག་༡བཅུ༢བརྒྱ༣་སྟོང་༤ཁྲི༥འབུམ་༦དང་། ས་ཡ༧་བྱེ་བ༨དུང་ཕྱུར༩ཏེ། དགུ་པོ་རྐྱང་བའི་གྲངས་ཡིན་ནོ།། ཐེར་འབུམ་༡ཁྲག་ཁྲིག༢གཏམ་པ༣དང་། གཏམ་དཀྲིག༤མུ་མིག༥མཐའ་འདས་༦དང་། དཔག་མེད་༧་བརྗོད་འདས༨ཁྱད་ཕྱིན༩དང་། འཕར་བ་༡༠་འདྲེན་པ་༡༡མཐར་སྣང་༡༢དང་། རྒྱུ་༡༣དང་འབྲས་བུ༡༤ལེགས་འབྱོར་༡༥་དང་། དབང་པོ་༡༦རྟོགས་པ་༡༧དཔྱོད་པ་༡༨དང་། ཚོལ་བ་༡༩དང་ནི་གྲངས་ཀྱི་མིག༢༠། གྲངས་ཀྱི་བརྡའ༢༡དང་རྣམ་པ་༢༢དང་། དཔེ་མཚོན་༢༣་མཐོ་སྟོན་༢༤བརྗོད་མི་ལངས་༢༥། དེ་དག་རྣམས་ནི་ཆེན་པོར་བཅས། ལྔ་བཅུ་ཁུག་པའི་གྲངས་ཡིན་ཞིང་། གྲངས་མེད་པ་༦༠ནི་ཐ་མ་སྟེ། བཅུ་འགྱུར་རིག་པས་བསྒྲེ་བ་ཡིན། དེ་ནས་མིང་གི་མངོན་བརྗོད་ལ། རྒྱས་པར་བརྗོད་ཀྱིས་མི་ལངས་ཀྱང་། རྩིས་ལ་ཉེར་མཁོ་མདོར་བསྡུས་ཙམ། བཤད་ཀྱིས་དོན་གཉེར་རྣ་བར་སློབས། ཟླ་གཟུགས་རི་བོང་བསེ་རུ་དང་། འོད་དཀར་ཅན་རྣམས་གཅིག་གི་མིང་། ལག་མིག་ཟུང་འཁྲིག་བགྲོད་སྦྱོར་དང་། མཚེ་མ་ཐབས་ཤེས་གཉིས་ཀྱི་མིང་། འཇིག་རྟེན་ཚ་བ་རྩེ་མོ་མེ། ཡོན་ཏན་སྲིད་པ་གསུམ་གྱི་མིང་། རྒྱ་མཚོ་ཆུ་བོ་ཆུ་གཏེར་དང་། སྡེ་བ་རིག་བྱེད་ཟུང་ལྡན་བཞི། མདའ་དང་འབྱུང་བ་དབང་པོ་དང་། འདོད་ཡོན་ཉེར་སྤྱོད་ཕུང་བོ་ལྔ། རོ་དུས་མཚམས་དང་བྲོ་བ་དང་། རྒྱན་དང་བཟང་པོ་དྲུག་གི་མིང་། རིན་ཆེན་ཐུབ་པ་དྲང་སྲོང་དང་། གཟའ་དང་རི་བོ་བདུན་གྱི་མིང་། བཀྲ་ཤིས་ལྷ་ཀླུ་ནོར་གདེངས་ཅན། ལྟོ་འགྲོ་སྦྲུལ་དང་དབྱིག་རྣམ་བརྒྱད། རྩ་གཏེར་སྲིན་པོ་བུ་ག་དགུ། ཁྲོ་བོ་འབྱོར་བ་སྟོབས་ཕྱོགས་བཅུ། དབང་ཕྱུག་དྲག་པོ་བདེ་འབྱུང་དང་། བྱེད་པ་བཅུ་གཅིག་མིང་ཡིན་ནོ།། རྟེན་འབྲེལ་ཉི་མ་ཁྱིམ་བཅུ་གཉིས། ལུས་མེད་མྱོས་བྱེད་འདོད་པ་དང་། གདུག་རིམ་སྣ་ཚོགས་བཅུ་གསུམ་གྲགས། སྲིད་གནས་མ་ནུ་ཤེད་བུ་དང་། ཡིད་རྣམས་བཅུ་བཞིའི་མིང་དུ་བཤད། ཚེས་དང་ཉིན་ཞག་བཅོ་ལྔ་ལ། མི་བདག་རྒྱལ་པོ་བཅུ་དྲུག་གོ། ཞེས་སྐྱོན་ཁམས་རྣམས་བཅོ་བརྒྱད་མིང་། རྒྱལ་བ་ཡུལ་ནི་ཉེར་བཞིར་བརྗོད། དེ་ཉིད་ཞེས་པ་ཉེར་ལྔའོ།། འཁོར་ལོ་སྐར་མ་ཉེར་བདུན་མིང་། སོ་དང་གཉིས་སྐྱེས་སུམ་ཅུ་གཉིས། སྟོང་པ་ནམ་མཁའ་བོན་དབྱིངས་ཐིག། གཞན་རྣམས་དེ་ཡི་འགྲོས་སུ་ཤེས། དེ་ནས་ལོ་ཡི་ཐ་སྙད་བཤད། ཁྱིམ་ཟླ་བཅུ་གཉིས་བགྲོད་གཉིས་དང་། ཉི་མས་རྒྱུ་སྐར་འཁོར་བ་དང་། དཔྱིད་དབྱར་སྟོན་དགུན་བཞི་རྫོགས་ཚེ། ལོ་གཅིག་ཞེས་པའི་ཐ་སྙད་གདག། རང་སྐྱོབས་འདོར་བྱེད་བགྲང་གཞི་དང་། དུས་ཀྱི་ཚིགས་དང་ཟླ་བའི་ཕྲེང་། བསྒྲང་བྱ་དུས་ཀྱི་ཆ་ཤས་ཅན། མི་ཟད་པ་རྣམས་ལོ་ཡི་མིང་། བྱི་གླང་སྟག་ཡོས་འབྲུག་སྦྲུལ་དང་། རྟ་ལུགས་སྤྲེལ་བྱ་ཁྱི་ཕག་སྟེ། ཕོ་ལོ་མོ་ལོ་གཉིས་གཉིས་ཏེ། ཤིང་མེ་ས་ལྕགས་ཆུ་ལྔ་ཡིས། འབྱུང་བས་རིམ་བཞིན་འགྲེས་པ་སྟེ། གོ་རྒན་དྲུག་ཅུར་གྲགས་པ་ཡིན། དེ་ཡི་མིང་འདོགས་མང་ན་ཡང་། གཡེ་ཁམས་སེལ་བིའ་མདོ་སྡེ་ལས། བྱུང་བའི་བརྡ་མིང་འདི་ལྟར་ཏེ། ཤིང་ཕོ་བྱི་བ་ནས་བཟུང་བའི། འབྱུང་ལྡན་རྫོགས་པ་རབ་འབར་དང་། སྦྱིན་འགྱུར་དཔའ་བོ་མྱོང་པ་དང་། ཀུན་རྫོགས་འབྲས་བདག་མཐོང་ལྡན་དང་། མཆོག་ཅན་ཕན་བདེ་ཀུན་དྷ་དང་། འབྲས་སྐྱེད་རུ་ལྡང་ས་འཛིན་དང་། མྱོས་རྒྱལ་རབ་ཀུན་མཚོན་པ་ཅན། རྟ་སྣོད་བདག་མཆོག་སྤྲིན་ཤུགས་དང་། བཟོད་དཀའ་ཐར་ལྡན་འཚོ་བ་དང་། རབ་རྒྱལ་ཤིང་འབྲས་དཔའ་སྐད་ཅན། གཏེར་འཛིན་སྡེར་འཛིན་གློག་ཅན་དང་། རྔ་མ་ཅན་དང་འདོད་ཆགས་འཛིན། དབང་ལྡན་བྲལ་བ་དགེ་བྱེད་དང་། སྡོས་ཅན་མཆེ་རྣོན་རྔ་སྒྲ་ཅན། འདོད་འབར་འཁྱིལ་ཅན་ཀུན་དགའ་བ། སྒྲ་འཛིན་གསུམ་པ་བདེ་ལྡན་དང་། ཕ་འཁོར་སྤྲིན་ཆེན་སྟུག་པོ་དང་། ཝ་བོང་དང་ནི་འཁྱིལ་པ་ཅན། ཟོ་བ་འདབ་རྒྱས་རི་བོ་དང་། དཔལ་ཅན་ཐུབ་པ་དབང་པོ་དང་། བུམ་འཛིན་ཀླུ་རལ་ཐུག་པ་དང་། ས་རལ་ཐུག་མོ་དང་བཅས་སོགས། ཆོས་ལུགས་བདེ་མཆོག་སྟོད་འགྲེལ་ལས། མེ་ཡོས་རབ་བྱུང་དོན་གཅིག་ནས། རིམ་བཞིན་ལོ་མིང་འདི་ལྟར་ཏེ། རབ་བྱུང་རྣམ་བྱུང་དཀར་པོ་དང་། རབ་སྨྱོས་སྐྱེས་བདག་ཨང་གི་ར། དཔལ་གདོང་དངོས་པོ་ན་ཚོད་ལྡན། འཛིན་བྱེད་དབང་ཕྱུག་འབྲུ་མང་པོ། མྱོས་ལྡན་རྣམ་གནོན་ཁྱུ་མཆོག་དང་། སྣ་ཚོགས་ཉི་མ་ཉི་སྒྲོལ་བྱེད་ས་སྐྱོང་མི་ཟད་ཐམས་ཅད་འདུལ། ཀུན་འཛིན་འགལ་བ་རྣམ་འགྱུར་དང་། བོང་བུ་དགའ་བ་རྣམ་རྒྱལ་ཉིད། རྒྱལ་བ་མྱོས་བྱེད་གདོང་ངན་པ། གསེར་འཕྱང་རྣམ་འཕྱང་སྒྱུར་བྱེད་དང་། ཀུན་ལྡན་འཕར་བ་དགེ་བྱེད་དང་། མཛེས་བྱེད་ཁྲོ་མོ་སྣ་ཚོགས་དབྱིག། ཟིལ་གནོན་སྤྲེའུ་ཕུར་བུ་དང་། ཞི་བ་ཐུན་མོང་འགལ་བྱེད་དང་། ཡོངས་འཛིན་བག་མེད་ཀུན་དགའ་དང་། སྲིན་པོ་མེ་དང་དམར་སེར་ཅན། དུས་ཀྱི་ཕོ་ཉ་དོན་གྲུབ་དྲག། བློ་ངན་རྔ་ཆེན་ཁྲག་སྐྱུག་པ། མིག་དམར་ཁྲོ་བོ་མི་ཟད་པའོ།། ལོ་ཡི་ཆ་ཤས་ལོའི་ཡན་ལག། ཉིན་ཞག་ཚོགས་དང་ལོ་ཡི་མཛོད། ཞག་གི་ཕུང་བོ་དུས་ཟླའི་མིང་། དྲོད་ཕབ་དུས་དང་མེ་ཏོག་ལྡན། མྱུ་གུའི་དུས་དང་བདེ་བྱེད་དུས། ངེས་སྲེག་ཚད་ཟླ་ཚ་གདུང་ཅན། དྲོད་ལྡན་སོ་ག་དཔྱིད་ཀྱི་མིང་། ཆར་ལྡན་ཆར་དུས་སྤྲིན་གྱི་དུས། སྤྲིན་འབྱུང་སྤྲིན་སྐྱེ་དབྱར་གྱི་མིང་། འབྲས་ལྡན་འབྲས་སྨིན་ནགས་སྐྱེད་དང་། གསེར་ལྡན་མ་ནི་སྟོན་གྱི་མིང་། ཁ་བིའ་དུས་དང་གངས་ཀྱི་དུས། བསིལ་ལྡན་མ་ནི་དགུན་གྱི་མིང་། ཟླ་བ་སོ་སོའི་དབྱེ་བ་ནི། ཧོར་ཟླ་དང་པོ་ནས་བཟུང་སྟེ། ཞང་ཞུང་སྐད་དང་ཧོར་ཟླའི་གྲངས། རྒྱ་གར་སྐད་དང་ཁྱིམ་ཀྱི་མིང་། རྒྱ་ནག་སྐད་དང་རྒྱ་རྩིས་ལུགས། དུས་བཞི་ར་འབྲིང་གང་ཡིན་དང་། རྟེན་འབྲེལ་ཉ་སྐར་རྟགས་མིང་བཅས། རྣམ་གྲངས་བཅུའི་ཀུན་ཀ་བཤད། ཞང་སྐད་ཟླ་རི་ཏིག་ཞི་ཞེས། ཧོར་ཟླ་དང་པོའི་རྒྱ་གར་གྱིས། མ་གྷ་ཞེས་པ་བུམ་པའི་ཁྱིམ། རྒྱ་ནག་སྐད་དུ་དབྱི་ཡོལ་ཞེས། རྩིས་རྙིང་སྟག་ཟླ་དཔྱིད་ར་སྟེ། རྟེན་འབྲེལ་འདུ་བྱེད་རྟ་པ་ཉ། ཆོ་འཕྲུལ་དཀའ་ཐུབ་ཟླ་ཞེས་གྲགས། ཟླ་རི་ནེ་སུམ་ཧོར་ཟླ་གཉིས། ཕལྒུ་ནར་ཞེས་ཉ་ཡི་ཁྱིམ། དབྲི་ཡོལ་ཡོས་ཟླ་དཔྱིད་འབྲིང་པོ། རྟེན་འབྲེལ་རྣམ་ཤེས་དབོ་ཡིས་ཉ། འབྲས་ལྡན་ཟླ་བ་ཞེས་ཀྱང་བཤད། སུམ་ནེ་ཟླ་རི་ཧོར་ཟླ་གསུམ། ཙི་ཏྲི་ཞེས་པ་ལུག་གི་ཁྱིམ། གནས་ཡོལ་འབྲུག་ཟླ་དཔྱིད་ཐ་མ། རྟེན་འབྲེལ་མིང་ཟླ་ནག་པ་ཉ། སྦྲང་ཟླའམ་དྲི་ཞིམ་ལྡན་དུ་གྲགས། ཟླ་རི་བིང་ང་ཧོར་ཟླ་བཞི། བཱེ་ཤ་ཁ་ཞེས་གླང་གི་ཁྱིམ། གསི་ཡོལ་སྦྲུལ་ཟླ་དབྱར་ར་སྟེ། རྟེན་འབྲེལ་སྐྱེ་མཆེད་ས་ག་ཉ། མེ་ཏོག་ལྡན་ནི་རུས་སྦལ་ཟླ། ཟླ་རི་ང་དྲུག་ལྔ་པ་སྟེ། ཛྱེཥྠ་ཞེས་པ་འཁྲིག་པའི་ཁྱིམ། འུ་ཡོལ་རྟ་ཟླ་དབྱར་འབྲིང་སྟེ། རྟེན་འབྲེལ་རེག་པ་སྣྲོན་གྱི་ཟླ། ངེས་སྲེགས་སམ་འདོད་འཕེལ་ཟླ་ཞེས་གྲགས། ཟླ་རི་དྲུག་སྣི་དྲུག་པ་སྟེ། ཨ་ཥཱ་དྷ་ཞེས་ཀར་ཊའི་ཁྱིམ། བླུ་ཡོལ་ལུག་ཟླ་དབྱར་ཐ་མ། རྟེན་འབྲེལ་ཚོར་བ་ཆུ་སྟོད་ཟླ། གཙང་མའི་ཟླ་བ་ཞེས་སུ་འདོད། ཟླ་རི་སྣིས་སམ་བདུན་པ་སྟེ། ཤྲ་བ་ཎ་ཞེས་སེང་གེའི་ཁྱིམ། ཚིལ་ཡོལ་སྤྲེལ་ཟླ་སྟོན་ར་བ། རྟེན་འབྲེལ་སྲེད་པ་གྲོ་ཞུན་ཉ། མྱོས་ཟླའམ་སྨེ་བའི་ཟླ་བར་བཤད། ཟླ་རི་སྣི་གྱད་བརྒྱད་པ་སྟེ། བྷ་ཏྲ་པ་ན་བུ་མོའི་ཁྱིམ། སཱ་ཡོལ་བྱ་ཟླ་སྟོན་འབྲིང་པོ། རྟེན་འབྲེལ་ལེན་པ་ཁྲུམ་གྱི་ཟླ། ནོར་ལྡན་སྤྲིན་བཟང་ཅན་དུ་གྲགས། ཟླ་རི་གུ་དུན་དགུ་པ་སྟེ། ཨཤྦི་ནི་ཞེས་སྲང་གིས་ཁྱིམ། སྒྱིའུ་ཡོལ་ཁྱི་ཟླ་སྟོན་ཐ་མ། རྟེན་འབྲེལ་སྲིད་པ་དབྱུག་པ་ཉ། ལྟད་མོའི་ཟླ་བའི་རྩི་ཏོག་ཟླ། ཟླ་རི་ཅུ་ཙེ་བཅུ་པ་སྟེ། ཀརྚ་ཀ་དང་སྡིག་པའི་ཁྱིམ། ཤྲི་མོལ་ཕག་ཟླ་དགུན་ར་བ། རྟེན་འབྲེལ་སྐྱེ་བ་སྨིན་དྲུག་ཉ། རྨི་ལམ་ཅན་གྱི་ཟླ་བར་འདོད། ཟླ་རི་ཅུ་ཏིག་བཅུ་གཅིག་པ། མྲཱྀ་ག་ཤིརྵ་གཞུ་ཡི་ཁྱིམ། ཤཱི་དབྱི་ཡོལ་དང་བྱི་བའི་ཟླ། དགུན་འབྲིང་རྟེན་འབྲེལ་རྒ་ཤིའི་ཟླ། མགོ་ཉ་སྨལ་པོ་ཟླ་བ་དང་། ས་ཡི་དབང་ཕྱུག་གམ་ནུས་ལྡན་ནོ།། ཟླ་རི་བཅུ་ནི་བཅུ་གཉིས་པ། སོ་སོ་ཥྱ་ཞེས་དང་ཆུ་སྲིན་ཁྱིམ། ཤཱི་དབྱི་ཡོལ་དང་གླང་གི་ཟླ། དགུན་ཐ་རྟེན་འབྲེལ་མ་རིག་པ། ནུས་ཟླའམ་འཕེལ་བའི་ཟླ་བར་གྲགས། རྩིས་གསར་འདོད་ན་བཅུ་གཉིས་པ། དཔྱིད་ར་སྟག་ཟླ་ངོས་འཛིན་ནས། རིམ་བཞིན་བསྐོར་བར་ཤེས་པར་འགྱུར། དཀར་ཕྱོགས་ཞི་དང་རྒྱས་པའི་ཕྱོགས། འཕེལ་ཕྱོགས་གང་ཕྱོགས་ཕྱོགས་ལྔ་མ། ཆ་ཤས་དར་དང་ཟླ་བའི་སྟོད། ཕྱོགས་ཀྱིས་དང་པོ་ཡར་ངོའི་མིང་། དམར་ཕྱོགས་དབང་དང་དྲག་པོའི་ཕྱོགས། ནག་ཕྱོགས་འབྱུང་འདུ་དམ་ཅན་འདུ། འགྲིབ་ཕྱོགས་ཟླ་སྨད་ཕྱོགས་ཕྱི་མ། ཆ་ཤས་འཛད་རྣམས་མར་ངོའི་མིང་། དེ་ལ་དགའ་བ་བཟང་པོ་དང་། རྒྱལ་བ་སྟོངས་པ་རྫོགས་པའམ། ཨཱ་ཨཱི་ཨུའུ་ཨེ་ཨཱཽ་ལྔ་དང་། བྱིས་གཞོན་ལང་ཚོ་སྨིན་ནུབ་པ། གཟུགས་སྒྲ་དྲི་རོ་རེག་བྱ་སྟེ། ཟླ་བ་སོ་སོའི་ངོ་རེ་ལ། ལྔ་ཚན་གསུམ་དུ་འཁོར་བའོ།། ནམ་མཁའ་རླུང་མེ་ཆུ་དང་ས། དཀར་པོའི་ཕྱོགས་ལ་ལན་གསུམ་དང་། ས་ཆུ་མེ་རླུང་ནམ་མཁའ་ལྔ། ནག་ཕྱོགས་ལན་གསུམ་འཁོར་བར་བྱེད། གཟའ་ཡི་མིང་གི་རྣམ་གྲངས་ནི། སྣང་བའི་མུ་ཁྱུད་འཇིག་རྟེན་མིག། སྲིད་པའི་སྒྲོན་མེ་ནམ་མཁའི་ནོར། སྣང་བྱེད་འོད་བྱེད་ཉིན་མོར་བྱེད། འོད་ལྡན་མུན་སེལ་གཟའ་ཡི་བདག། རྟ་ལྗང་ཅན་དང་འོད་སྟོང་ལྡན། རྟ་བདུན་པ་དང་བདུན་གྱི་འདུན། ཚ་ཟེར་ཅན་དང་མི་བྱིན་སྐྱེས། སྣ་ཚོགས་ཤིང་རྟ་པད་མའི་གཉེན། མུན་པའི་དགྲ་བོ་རྟག་ཏུ་འཆར། ལོ་ཡི་ཤིང་རྟ་བཅུ་གཉིས་བདག། འགྲོ་བའི་སྒྲོན་མ་ཀུ་མུད་དགྲ། ནམ་མཁའི་ཏོག་རྣམས་ཉི་མར་གྲགས། བསིལ་ཟེར་ཅན་དང་བདུད་རྩིའི་འོད། ཆགས་བྱེད་འོད་དཀར་རྒྱ་མཚོ་དགའ། ཚེས་བདག་སིལ་བྱེད་རི་བོང་ཅན། བསིལ་ལྡན་འཚོ་བྱེད་ག་པུར་འཛིན། དྲག་པོའི་གཙུག་རྒྱན་རྒྱ་མཚོར་སྐྱེས། རྒྱ་མཚོའི་མར་གསར་པད་མའི་དགྲ། རི་བོང་འཛིན་སོགས་ཟླ་བའི་མིང་། གཞོན་ནུ་ལུས་དམར་ཡན་ལག་དམར། བཀྲ་ཤིས་ས་སྐྱེས་ས་ཡི་བུ། འོད་ཟེར་དགུ་པ་ཆུ་སྟོད་སྐྱེས། དམག་དཔོན་རྣམས་ནི་མིག་དམར་མིང་། མཁས་པ་ཟླ་སྐྱེས་ཟླ་བའི་བུ། ལུས་གཅིག་པ་དང་སྣར་མ་སྐྱེས། སྔོ་སངས་ལུས་དང་མཛའ་ལྷའི་གྲོགས། ལྷ་མཚམས་སྐྱེས་རྣམས་ལྷག་པར་གྲགས། སྣ་ཚོགས་གཙུག་ཅན་ལྷའི་བླ་མ། སྒྲ་མཁན་སྒྲ་མཁས་ལྷའི་སློབ་དཔོན། བློ་ལྡན་ཚིག་བདག་མེ་བཞི་སྐྱེས། རིག་བྱེད་བདག་པོ་ཕུར་བུའི་མིང་། ལྷ་མིན་བླ་མ་ངན་སྤོང་བུ། སྙན་ངག་མཁན་པོ་མཆུ་ལས་སྐྱེས། ཁུ་བ་དང་ནི་དཀར་པོ་དང་། དྲང་སྲོང་པ་ཝ་སངས་ཀྱི་མིང་། དལ་འགྲོ་ཉི་སྐྱེས་ཉི་མའི་བུ། འོད་ཟེར་བདུན་པ་མི་བཟད་མིག། ཚངས་སྐྱེས་འཁྱོགས་འགྲོ་རྗེས་སུ་སྐྱེས། གོས་སྔོན་ཅན་རྣམས་སྤེན་པར་འདོད། ཁམས་གསུམ་རྣམ་རྒྱལ་མུན་པ་ཅན། མཐོ་རིས་སྣང་བྱེད་ཟླ་བའི་དགྲ། མགོ་ཟླུམ་མི་ནག་གཟའི་རྣ་བ། བསོད་ནམས་ལྡན་རྣམས་སྒྲ་གཅན་མིང་། མིག་དམན་མདོག་ཅན་ཉི་ཟླ་བཟའ། དུས་ཀྱི་མེ་དང་སྐྱང་ཀི་དང་། སྣ་ཚོགས་མདོག་ཅན་ལ་སོགས་པ། མཇུག་མའི་མིང་དུ་ཤེས་པར་བྱ། སྐགས་ལ་སྐྱེས་དང་སྦྲུལ་ཅན་དང་། གཙུག་ཕུད་ཅན་དང་སྐྲ་གཉིས་པ། ཕོད་ཅན་དང་ནི་མཇུག་ཕོད་ཅན། དུ་བ་མཇུག་རིངས་མིང་དུ་གྲགས། གཞན་ཡང་གཡེན་ཁམས་སེལ་མདོ་ལས། གསུངས་པའི་གཟའ་བདུན་བརྡའ་མིང་ནི། རྟ་བདུན་༡་ཀུན་དྷ༢གཏུམ་པོ་༣དང་། མྱོས་ཅན༤དྲང་སྲོང༥དཔའ་བོ༦དང་། འཚུབ་པ༧ཞེས་དང་བདུན་དུ་བཤད། ཝེག་ཞི་ཞེས་པ་སྐར་མ་སྟེ། དབྱེ་ན་དུང་ཕྱུར་གཉིས་དང་ནི། བྱེ་བ་བརྒྱད་དང་ས་ཡ་ལྔ། ཡོད་པར་ལུང་ལས་བཤད་ན་ཡང་། དབང་སྒྱུར་རྒྱུ་སྐར་ཉེར་བདུན་ནམ། གྲོ་བཞིན་དང་ནི་བྱི་བཞིན་གཉིས། སོ་སོར་འཕྲལ་བས་ཉེར་བརྒྱད་དེ། དེ་རྣམས་མིང་འདོགས་རྣམ་གྲངས་དང་། གཡེན་ཁམས་སེལ་མདོ་ལས་བཤད་པའི། བརྡ་མིང་དང་བཅས་མདོར་བསྡུས་ན། ༢ སྨིན་དྲུག་ལ་ནི་མིང་དྲུག་དང་། བརྡའ་མིང་གྲུབ་རྒྱལ་ཞེས་སུ་བཤད། ༣ སྣར་མར་བིས་རྫི་རི་རབ་དང་། ༤ མགོ་ལ་སྨལ་པོ་ལྗང་ལོ་ཅན། ༥ ལག་ལ་ལག་མིང་ཞེས་སུ་གྲགས། ༦ ནབ་སོར་རྒྱལ་སྟོད་དཔག་བསམ་ཞེས། ༧ རྒྱལ་ལ་རྒྱལ་སྨད་བུམ་འཛིན་ནོ། ༨ སྐག་ལ་ཝ་དང་ཕུར་སོ་ཅན། ༩ མཆུ་ལ་རྟ་ཆེན་རྟ་མཆོག་དང་། ༡༠ གྲེ་ལ་རྟ་ཆུང་དང་གྲུ་འབུམ། ༡༡ དབོ་ལ་ཁྲ་དང་སྒྲོ་ཅན་ཞེས། ༡༢ མེ་བཞིར་བྱ་མ་དང་སྣང་སེ། ༡༣ ནག་པར་བྱའུ་དང་སྒོང་འཛིན། ༡༤ ས་རི་ནམ་གོང་མེ་རླུང་ཅན། ༡༥ ས་ག་ནམ་འགོ་དང་རླུང་རྒྱལ། ༡༦ ལྷ་མཚམས་ལག་སོར་ལྕགས་སྟོ་ཞེས། ༡༧ སྣྲོན་ལ་གདུབ་བུ་སྟེའུ་ཅང་། ༡༨ སྣྲུབ་ལ་གྲུ་མོ་དང་དཔུང་སོགས། བརྡའ་མིང་ཟུར་རྒྱལ་ཞེས་སུ་གྲགས། ༡༩ བྲེ་ལ་ཆུ་སྟོད་དང་འབྲས་ཅན། ༢༠ ཕུལ་ལ་ཆུ་སྨད་དང་ས་བདག། ༢༡ གྲོ་བཞིན་ལ་ནི་བྱ་སྦོ་དང་། མོན་དང་མཆུ་རིང་ཞེས་སུ་བཤད། ༢༡ བྱི་ཞིན་ལ་ནི་བྱ་གཞུག་དང་། སྤུར་དང་ཆུ་རྒྱལ་ཞེས་སུ་གྲགས། ༢༢ མོན་འདྲེ་སྒྲོག་སྟོད་དང་རླུང་འཛིན། ༢༣ མོན་གྲུར་སྒྲོག་སྨད་དང་གསུམ་ལྡན། ༢༤ ཁྲུམ་སྟོད་སེ་སྟོད་དང་དགྲ་མང་། ༢༥ ཁྲུམ་སྨད་སེ་སྨད་བརྩེ་འཛིན་ཅན། ༢༦ ནམ་གྲུར་ཤེ་ས་གར་བ་ཞེས། ༢༧ ཐ་སྐར་དབྱུག་པ་དང་ཡུད་རིང་། ༡ བྲ་ཉེར་བརྡའ་མིང་ནག་པའོ།། གཡེན་ཁམས་མདོ་ལས་བྱུང་བ་ཡི། ཚེས་གཅིག་ནས་བཟུང་གནམ་སྟོང་བར། ཚེས་ལྷ་སུམ་ཅུ་འཆར་བ་སྟེ། ཆ་ལུགས་དང་བཅས་རིམ་བཞིན་བཤད། ༡ མང་གྱུར་དཀར་པོ་རྒོད་གཤོག་འཛིན། ༢ ཆགས་བྲལ་མཐིང་ཁ་སྒོང་རྡོ་སྒྱུར། ༣ དགའ་ལྡན་སེར་པོར་བུམ་པ་འཛིན། ༤ འབྲས་མཆོག་དམར་པོ་རེའུ་མིག་འཛིན། ༥ མྱོང་བ་སྔོ་སྐྱ་ཞགས་པ་འཛིན། ༦ གྲུ་འཛིན་སེར་པོ་ཤིང་རྟ་སྒྱུར། ༧ ཟུག་བྲལ་ལྗང་སྐྱ་ཤིང་ལོ་འཛིན། ༨ དཀར་པོ་ཟླ་དཀར་རྒྱས་པ་འཛིན། ༩ མཆོག་ཆེན་སྨུག་པོ་མྱུ་གུ་འཛིན། ༡༠ བདེ་བ་སེར་པོ་གཏོར་གཞོང་འཛིན། ༡༡ ཞབས་ལྡན་དཀར་པོ་ཧོས་རུ་འཛིན། ༡༢ འཁོར་ལོ་དཀར་པོ་འཁོར་ལོ་འཛིན། ༡༣ ཞལ་རྒྱས་དཀར་པོ་གླིང་བུ་ཅན། ༡༤ སད་འཛིན་དཀར་པོ་འདབ་རྒྱས་འཛིན། ༡༥ རྒྱས་ཅན་དཀར་པོ་ཉི་མ་འཛིན། ༡༦ སྒྲ་བྲལ་དཀར་པོ་ཞགས་པ་འཛིན། ༡༧ ཉུངས་ཀར་སེར་པོ་འབྲས་སྙིང་འཛིན། ༡༨ ཡིག་ཚང་དམར་པོ་གབ་རྩེ་འཛིན། ༡༩ དཀར་ཞལ་ལྗང་ཁུ་ལྔ་ཆེན་བརྡུང་། ༢༠ ཚེ་སྐྱོངས་དམར་པོ་དུང་དཀར་འབུད། ༢༡ འཕྲེང་ཅན་དམར་སེར་ཕྲེང་བ་འཛིན། ༢༢ ངང་སྐྱལ་ཞོ་པོ་ཆུ་ཞགས་འདྲེན། ༢༣ མཚམས་འཛིན་སེར་པོ་རི་རབ་འཛིན། ༡༤ ཆུ་བོ་སྔོན་པོ་རྒྱལ་མཚན་འཛིན། ༢༥ རི་དབང་ལྗང་གུ་རྡོ་རྗེ་འཛིན། ༢༦ འཐོར་རེ་སེར་པོ་བེ་ཅོན་འཛིན། ༢༧ གླང་མཆོག་སྔོན་པོ་མཆེ་བ་འཛིན། ༢༨ གྷན་དྷེ་སྨུག་པོ་གྷན་དྷེ་འཛིན། ༢༩ ཉ་འཛིན་སྔོན་པོ་མདའ་གཞུ་འཛིན། ༣༠ གཞིང་ལྡན་མཐིང་ཁ་པད་གཞོང་འཛིན། ཚེས་བདག་རྣམས་ནི་གནོད་སྦྱིན་ཚུལ། ཉིན་ཞག་དབང་བའི་ལས་བྱེད་པས། སོ་སོའི་བྱ་བ་སྤང་བླང་བསྒྲུབ། དེ་དག་གནས་བཞིན་ཡོན་ཏན་འབྱུང་། གཞན་ཡང་གཡེན་ཁམས་སེལ་མདོ་ལས། རྒྱན་སྦྱོར་ཞེས་པ་བདུན་པོ་ནི། མཛེས་མ་ཀླུང་དང་རོལ་མཚོ་དང་། བུ་ག་ཤོལ་མདའ་བུང་བ་དང་། སྒྲ་འཛིན་ཞེས་བྱ་བརྗོད་པ་སྟེ། མཛེས་མ་བུད་མེད་མེ་ལོང་འཛིན། ཀླུ་རྒྱལ་མེ་ཏོག་ཕྲེང་བ་འཛིན། རོལ་མཚོ་ཆུ་བོ་རྒྱུན་འབབ་སྟེ། བུ་ག་ཆོ་ལོ་མིག་དགུ་བ། ཤོལ་མདའ་ཤིང་ལས་གྲུབ་པ་སྟེ། བུང་བ་སྟག་སྦྲང་གཤོག་པ་ཅན། སྒྲ་འཛིན་རྒྱུད་མང་ལྡིར་བ་བཅས། དེ་དག་ཚེས་ལ་རྒྱུ་བ་སྟེ། དཔྱིད་ར་དང་ནི་སྟོན་འབྲིང་པོའི། ཚེས་གཅིག་མཛེས་མ་ནས་ཚེས་ཤིང་། དཔྱིད་འབྲིང་སྟོན་ཐ་མཚོའི་ཚེས། དཔྱིད་ཐ་དགུན་རར་ཤོལ་མདའ་དང་། དབྱར་ར་དགུན་འབྲིང་སྒྲ་འཛིན་དང་། དབྱར་འབྲིང་དགུན་ཐ་ཀླུ་མོའི་ཚེས། དབྱར་ཟླ་ཐ་ཆུང་བུ་ག་དང་། སྟོན་ར་བུང་བ་ནས་ཚེས་ཏེ། རིམ་བཞིན་བསྐོར་བས་འཁྲུལ་མེད་འགྱུར། དེ་ནས་མདོ་སྡེ་དེ་དག་ལ། བཤད་པའི་དུས་འབྲས་བཅུ་གཉིས་ནི། ཉི་མ་སེང་གེ་ཆུ་སྲིན་དང་། ཁྱུ་མཆོག་རིགས་ལྡན་བུ་མོ་དང་། མྱོས་རྒྱལ་འཁོར་ལྡན་དུས་བསྲུང་དང་། དུས་ཆེན་མ་རུང་སྒྲ་འཛིན་བཅས། དེ་ཡི་གནས་ཚུལ་མདོར་བསྡུས་ནི། ཟླ་བ་དང་ནི་ཉིན་ཞག་དང་། ཞག་གི་ཆ་ཤས་དབང་བ་སྟེ། དང་པོ་ཟླ་བར་དབང་ཚུལ་ནི། དཔྱིད་རར་ཉི་མ་དཔྱིད་འབྲིང་ལ། སེང་གེ་ལ་སོགས་རིམ་བཞིན་དབང་། གཉིས་པ་ཉིན་ཞག་དབང་ཚུལ་ནི། ཕོ་ཟླ་དྲུག་ལ་ཉི་མས་ཚེས། མོ་དྲུག་མྱོས་རྒྱལ་གྱིས་ཚེས་ཏེ། གནམ་སྟོང་བར་དུ་རིམ་བཞིན་བསྐོར། གསུམ་པ་དུས་ལ་དབང་ཚུལ་ནི། ཉིན་མཚན་དུས་ཚོད་བཅུ་གཉིས་ལ། ནམ་ལང་ཉི་མ་ཉི་ཤར་སེང་། དེ་སོགས་རིམ་བཞིན་ཤེས་པར་བྱ། གཟའ་སྐར་འཕྲོད་པའི་འཕྲད་མིང་ནི། ཀུན་དཀའ་དུས་དབྱིག་དུལ་སྐྱེ་དགུ། གཞོན་དང་བྱ་རོག་རྒྱལ་མཚན་དང་། དཔལ་བེའུ་རྡོ་རྗེ་ཐོ་བ་གདུགས། གྲོགས་ཡིད་འདོད་པ་འགལ་མེ་དང་། རྩ་བཏོན་འཆི་བདག་མདའ་གྲུབ་མདུང་། བདུད་རྩི་གཏུན་ཤིང་གླང་བོ་དང་། རྟག་མྱོས་ཟད་པ་གཡོ་བརྟེན་འཕེལ། སྦྱོར་བ་ཉེར་བདུན་ཞེས་བྱ་ནི། སེལ་བ་མཛའ་བོ་ཚེ་ལྡན་དང་། བསྐལ་བཟང་བཟང་པོ་ཤིན་ཏུ་སྐྲངས། ལས་བཟང་འཛིན་པ་ཟུག་རྔུ་སྐྲངས། འཕེལ་བ་ངེས་པ་ཀུན་འཇོམས་དང་། དགའ་བ་རྡོ་རྗེ་དངོས་གྲུབ་དང་། ཤིན་ལྟུང་མཆོག་ཅན་ཡོངས་སུ་འཇོམས། ཞི་བ་གྲུབ་པ་བསྒྲུབ་བྱ་དང་། དགེ་བ་དཀར་པོ་ཚངས་པ་དང་། དབང་པོ་ཁོན་འཛིན་ཉེར་བདུན་ནོ།། བྱེད་པ་བཅུ་གཅིག་ཐ་སྙད་ནི། གདབ་པ་བྱིས་པ་རིག་པ་ཅན། ཏིལ་བརྡུང་ཁྱིམ་སྐྱེས་ཚོང་བིའྚི། ཚེས་གཅིག་ཕྱི་ཆ་ནས་བརྩམས་ཏེ། ཉེར་དགུའི་སྔ་ཆའི་བར་དག་ཏུ། འཁོར་བས་འཕོ་བྱེད་བདུན་ཞེས་གྲགས། བཀྲ་ཤིས་དང་ནི་རྐང་བཞི་ཀླུ། མི་ཤུག་བཞི་ནི་ཉེར་དགུ་ཡི། ཕྱི་ཆ་ནས་བཟུང་ཚེས་གཅིག་གི། སྔ་ཆའི་བར་དུ་རིམ་ཤར་བས། རྟག་བྱེད་བཞི་ཞེས་གྲགས་པའོ།། རྟེན་འབྲེལ་བཅུ་གཉིས་འདི་ལྟར་ཏེ། མ་རིག་འདུ་བྱེད་རྣམ་པར་ཤེས། མིང་གཟུགས་སྐྱེ་མཆེད་རེག་པ་དང་། ཚོར་བ་སྲེད་པ་ལེན་པ་དང་། སྲིད་པ་སྐྱེ་བ་རྒ་ཤི་རྣམས། ཟླ་བའི་སྒང་ནས་སྒང་ཚད་བར། ཤོག་ཁྲ་བཞིན་དུ་རིམ་འཆར་བ། སྐར་རྩིས་དུས་འཁོར་ལུགས་བཞིན་ནོ།། གཞན་ཡང་གཙུག་ལག་ནག་རྩིས་ལས། བཤད་པའི་ཉི་མ་བཅུ་གཉིས་ནི། སྟག་ཉི་ནས་བཟུང་གླང་ཉི་བར། ཕོ་ཟླ་དྲུག་ལ་སྟག་ནས་ཚེས། མོ་དྲུག་སྤྲེལ་ནས་ཚེས་ཏེ་འཁོར། སྤར་ཁ་བརྒྱད་ནི་ལི་ཁོན་དང་། དྭ་ཁེན་ཁམ་གིང་ཟིན་ཟོན་བརྒྱད། སྟག་རྟ་ཁྱི་གསུམ་ལི་ནས་ཚེས། ཕག་ལུག་ཡོས་གསུམ་ཟིན་ནས་ཚེས། བྱི་འབྲུག་སྤྲེལ་གསུམ་ཁམ་ནས་ཚེས། བྱ་གླང་སྦྲུལ་གསུམ་དྭ་ནས་སོ།། སྨེ་བ་དགུ་ནི་གཅིག་དཀར་དང་། གཉིས་ནག་གསུམ་ཐོན་བཞི་ལྗང་དང་། ལྔ་སེར་དྲུག་དཀར་བདུན་དམར་དང་། བརྒྱད་དཀར་དགུ་སྨུག་རྣམས་ཡིན་ཏེ། ར་བ་བཞི་ལ་གཅིག་དཀར་དང་། འབྲིང་པོ་བཞི་ལ་བཞི་ལྗང་དང་། ཐ་ཆུང་བཞི་ལ་བདུན་དམར་ནི། ཚེས་ཏེ་རིམ་བཞིན་འཁོར་བའོ།། ཉིན་མཚན་དུས་ཚོད་བཅུ་གཉིས་ལ། ནང་ལངས་ཡོས་ནས་ཉི་ཤར་འབྲུག། ཉི་དྲོས་སྦྲུལ་ལ་ཉིན་ཕྱེད་རྟ། ཕྱེ་ཡོལ་ལུག་ལ་ཉི་མྱུར་སྤྲེལ། ཉི་ནུབ་བྱ་ལ་ས་སྲོས་ཁྱི། སྲོད་འཁོར་ཕག་ལ་ནམ་ཕྱེད་བྱི། ཕྱེད་ཡོལ་གླང་དང་ཐོ་རངས་སྟག། དེ་རྣམས་གཙུག་ལག་ནག་རྩིས་ལུགས། ཞེས་སྐར་རྩིས་ཀྱིས་ལེགས་པར་བཤད་པ་སྣང་གསལ་སྒྲོན་མ་ངོ་མཚར་འོད་ཟེར་འབུམ་ལྡན་ལས། ཉེར་མཁོ་སྔོན་འགྲོའི་རིམ་པ་མདོ་ཙམ་བསྟན་པའི་སྐབས་ཏེ་གཉིས་པའོ།།
