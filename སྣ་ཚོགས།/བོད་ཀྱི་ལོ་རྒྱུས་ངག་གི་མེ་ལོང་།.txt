 བདེན་དྲང་ཆོས་ཉིད་འཚོལ་བའི་རིགས་པ་དང་། །རིགས་པས་བདེན་དོན་གཉེར་བའི་རྣམ་དཔྱོད་ཀྱིས། །ལོ་རྒྱུས་དུང་གི་ཟླ་བར་བྱི་དོར་བའི། །ཡ་རབས་བློ་ལྡན་དག་ལ་གུས་པས་འདུད། །རང་གི་ཕ་ཡི་ངོ་གདོང་ཤེས་པའི་ཕྱིར་། །བྱུང་ནས་ཡལ་བའི་ལོ་རྒྱུས་གནས་ལ་སྦྱོང་། །འདི་ཉིད་མ་ཤེས་ནགས་ཀྱི་སྤྲེའུ་སྟེ། །ཕ་བཟང་བུ་ཡི་དོན་དུ་ཅུང་ཟད་འབྲི། །གངས་ལྗོངས་འདི་རུ་སྔོན་ཆད་མཚོ་ཡིས་ཁྱབ། །མཚོ་དེ་འཁོལ་བ་ཞི་ཞིང་རབ་དུ་བསིལ། །ཀོང་གི་ཆུ་བརླག་ཅེས་བྱའི་ཟོམས་ཕྱེས་པས། །རུ་བཞི་ཆུ་ཡིས་གང་བ་དེར་ཐིམ་པས། །བོད་ཁམས་ཡུལ་གྱི་རྣམ་པ་གསལ་བར་བཀོད། །མངའ་རིས་སྐོར་གསུམ་ཤ་རྐྱང་རི་དྭགས་གླིང་། །བར་གྱི་རུ་བཞི་སྟག་གཟིག་གཅན་གཟན་གླིང་། །སྨད་ཀྱི་སྒང་དྲུག་འདབ་ཆགས་བྱ་ཡི་གླིང་། །མཐའ་བཞི་གངས་རི་ར་བས་ཡོངས་སུ་བསྐོར། །སྤྲུལ་བའི་སྤྲེའུ་དང་སྲིན་མོ་འདུས་པ་ལས། །རིས་དྲུག་ནས་འཕོས་སྤྲེལ་ཕྲུག་དྲུག་ཏུ་བྱུང་། །དེ་ཀུན་ཕ་ཡིས་ཤིང་ཏོག་གིས་གང་བ། །ནགས་མ་བྱ་ཚོགས་ཅན་ཞེས་བྱ་བར་སྐྱེལ། །དེ་ལས་མི་མིན་སྤྲེལ་མིན་བཞི་བརྒྱ་འཕེལ། །ཤིང་ཏོག་ཟད་ཅིང་བཀྲེས་པས་འཐབ་ཅིང་རྩོད། །མ་རྨོས་ལོ་ཏོག་ཟོས་པས་མི་རུ་གྱུར། །མ་རྨོས་འབྲུ་ཟ་གོས་སུ་ཤིང་ལོ་གྱོན། །ཡུལ་ལ་སྔ་བ་ཡར་ཀླུངས་སོག་ཀ་རུ། །ཞིང་ལ་སྔ་བ་ཟོ་ཐང་ཞེས་པ་བྱུང་། །ནགས་ན་རི་དྭགས་ལྟ་བུར་གནས་པ་ཡིས། །ཀློ་དང་མོན་ལྟར་སྤྱོད་པས་བོད་ཡུལ་གང་། །ཕན་ཚུན་མ་མཐུན་འཐབ་པས་བཞིར་གྱེས་ཏེ། །རུས་ལ་སྔ་བ་སེ་རྨུ་སྟོང་ལྡོང་བྱུང་། །གཞན་ཡང་སྲིད་པའི་རྒྱལ་བོ་འཐིང་ཞེས་པར། །བུ་གསུམ་སྐྱེས་པས་རྒྱ་བོད་སོག་གསུམ་མཆེད། །བོད་མི་ཁྲི་དོ་ཆེན་པོ་བྱ་བ་ལ། །བུ་དྲུག་འཁྲུངས་པས་བོད་མི་གདུང་དྲུག་ཟེར། །ཕྱོགས་བཞི་རྒྱལ་བོ་ཕྱོགས་ནས་དར་བའི་ཚེ། །དེ་དུས་བོད་ཁམས་མི་ལ་རྗེ་མེད་པས། །རི་བྲན་བྲག་མཁར་ཀུན་ལ་རྫོང་ཡང་བཅས། །གོས་དང་ཁ་ཟས་བཙན་མཁར་ནོར་བུ་བཅུག །རྒོལ་དུ་མི་ཐུབ་དམག་གྲངས་ཉུང་བའི་དུས། །རང་ལ་དབང་མེད་རྒྱབ་ཚོགས་སྲབ་པའི་དུས། །བོད་ཡུལ་གངས་རིའི་རྩེ་དགུའི་ལྗོངས་འདི་རུ། །མཐའ་ཡི་རྒྱལ་བོ་བཞི་བོ་འདུལ་འདོད་ནས། །པུར་རྒྱལ་གོང་དུ་མངའ་མཛད་བཅུ་བོ་དག །རིམ་པར་བྱུང་ཡང་ཕྱོགས་བཞིའི་དགྲ་མ་ཐུབ། །དང་བོ་གནོད་སྦྱིན་ནག་པོས་དབང་བྱས་ཏེ། །ཡུལ་གྱི་མིང་ཡང་ཡུལ་བཟང་རྒྱན་མེད་ཟེར། །མཚོན་ཆ་མདའ་བཞི་འཕེན་པ་དེ་ནས་བྱུང་། །གཉིས་པ་རེ་ལྡེ་མགོ་གཡག་བདུད་ཀྱིས་བྱས། །མིང་ཡང་བདུད་ཡུལ་ཁ་རོག་རོང་དགུར་བཏགས། །དགྲ་སྟ་སྟ་རེ་དུས་དེར་ལག་ཆ་བྱུང་། །སྔ་རྟགས་གཉེན་པོ་བཙན་པ་དེ་ནས་བྱུང་། །གསུམ་པ་གཉའ་རིང་ཁྲག་མེད་སྲིན་གྱིས་བྱས། །ཡུལ་གྱི་མིང་ཡང་ནག་པོ་དགུ་དུལ་ཟེར། །ལག་ཆར་སྲིན་མོ་རྐང་དང་སྒྱོགས་གཉིས་བྱུང་། །བཞི་བ་འཇམ་དམར་ལྷ་ཡིས་དབང་བྱས་ཏེ། །ཡུལ་གྱི་མིང་ཡང་ལྷ་ཡུལ་གུང་ཐང་ཟེར། །ལག་ཆར་ཁ་ཊཾ་རྩེ་གསུམ་དེ་ནས་བྱུང་། །སྔ་རྟགས་ཤ་ཟ་གདོང་བ་དམར་བོ་བྱུང་། །ལྔ་བར་མུ་རྒྱལ་ཁོ་རྗེས་དབང་བྱས་ཏེ། །ཡུལ་གྱི་མིང་ཡང་ངམ་འཕྲང་ཅང་བྲང་ཟེར། །ལག་ཆར་མདུང་ལ་བྱེད་པ་དེ་ནས་བྱུང་། །དྲུག་པར་ཀྲོག་ཀྲོག་འདྲེ་ཡིས་དབང་བྱས་ཏེ། །ཡུལ་གྱི་མིང་ཡང་ལང་ཏང་ལིང་ཏང་ཟེར། །ལག་ཆར་འུར་རྡོ་འཕེན་པ་དེ་ནས་བྱུང་། །བདུན་པ་མ་སངས་རུས་དགུས་དབང་བྱས་ཏེ། །ཡུལ་གྱི་མིང་ནི་བོད་ཁ་ཉ་དྲུག་ཟེར། །ལག་ཆར་དོང་རལ་གོ་ཆ་ཕུབ་ཆུང་བྱུང་། །སྔ་རྟགས་མདའ་མདུང་མཚོན་ཆ་དེ་ནས་བྱུང་། །བརྒྱད་པ་ཀླུ་ཡིས་དབང་དུ་བྱས་པ་ལས། །ཡུལ་གྱི་མིང་ནི་བོད་ཁམས་གླིང་དགུ་བཏགས། །སྔ་རྟགས་དཀར་བུ་ཆུ་ལ་སྤྱོད་པ་བྱུང་། །དགུ་བ་མི་མ་ཡིན་གྱིས་དབང་བྱས་ཏེ། །ཡུལ་གྱི་མིང་ནི་ངམ་ཡུལ་ནག་པོར་བཏགས། །བཅུ་བ་ཟ་རིད་སྤུན་དགུས་དབང་བྱས་ཏེ། །ཡུལ་གྱི་མིང་ནས་སྟོང་སྡེ་བཅོ་བརྒྱད་བཏགས། །སྔ་རྟགས་རྟ་ཞོན་རྣ་ཆ་འདོགས་པ་བྱུང་། །དེ་ནས་རྒྱལ་ཕྲན་བཅུ་གཉིས་དབང་བྱས་ཏེ། །ཡུལ་གྱི་མིང་ནི་སྭ་ཁ་བརྒྱད་དུ་བཏགས། །སྔ་རྟགས་ཕྱག་དང་ཞེ་ས་དེ་ནས་བྱུང་། །ཀུན་གྱི་ཐ་མར་བཟངས་ཡུལ་གྱད་དུ་བཏགས། །དེ་ལ་སྟོད་ཀྱི་སིལ་མ་ཞེས་སུ་གྲགས། །རྒྱལ་ཕྲན་དེ་རྣམས་འཐབ་རྩོད་གསོད་ལ་དགའ། །ལེགས་ཉེས་མི་བརྩི་བཀྱོན་འབེབ་བཙོན་ལ་འཇུག། །མཐའ་བཞིའི་རྒྱལ་བོ་རྟག་ཏུ་མནར་ཞིང་གཙེས། །རྒྱ་ནག་རྒྱལ་བོ་ཤིང་ལ་སྦྲུལ་ལྟར་འཁྲི། །རྒྱ་གར་རྒྱལ་བོ་ལུག་ལ་སྤྱང་ལྟར་འཇབ། །སྟག་གཟིག་རྒྱལ་བོ་བྱ་ཁྱུར་ཁྲ་ལྟར་རྒྱུག། །གེ་སར་རྒྱལ་བོ་ཤིང་མ་སྟ་ལྟར་འཚབ། །འཐབ་ཏུ་མི་ཐུབ་བོད་ལ་དམག་གྲངས་ཉུང་། །ཡུལ་གཞན་འགྲོ་བར་རྒྱལ་ཕྲན་གྱིས་མི་སྟེར། །ཐང་ལ་མི་ཆགས་བྲག་རི་བཙན་པོ་འཛིན། །བཟའ་བཏུང་མི་རྙེད་བཀྲེས་སྐོམ་སྡུག་བསྔལ་ཆེ། །བོད་ཡུལ་མི་རྣམས་སྡུག་གིས་མནར་བའི་ཚེ། །མི་ཡི་རྗེ་བོ་གཉའ་ཁྲི་བཙན་པོ་བྱུང་། །དེ་ཡིས་བོད་འབངས་བདེ་ཞིང་སྐྱིད་པར་མཛད། །ལྒིང་ལུགས་མ་མཐུན་གསང་བསྒྲགས་ཡང་གསང་ཟེར། །གང་ལྟར་འཐེན་ཡང་གཉའ་ཁྲིར་མཐུན་པ་སྟེ། །བསྒྲགས་པ་བོན་ལུགས་མི་རྗེ་ལྷ་ཡིས་མཛད། །གསང་བ་ཆོས་ལུགས་རྒྱ་གར་རིགས་ནས་མཆེད། །ཡང་གསང་སྤོ་བོའི་ཐེ་རིང་རྒྱུད་པ་སྟེ། །གང་ལྟར་ལྷ་རི་གྱང་དོར་བབས་པའི་ཚེ། །ཡབ་འབངས་རུས་དྲུག་ཤེས་ཅན་བཅུ་གཉིས་ཀྱིས། །གཉའ་ཡི་ཁྲི་ལ་བཏགས་ནས་སྤྱན་དྲངས་པས། །མཚན་ཡང་གཉའ་ཁྲི་བཙན་པོ་ཞེས་སུ་གྲགས། །སུམ་པའི་བོན་པོ་ཨོ་ཡོང་རྒྱལ་བ་བཏུལ། །སྣུབས་ཀྱི་སྲིད་པ་རྒྱལ་ཕྲན་འབངས་སུ་འདུ། །ཡུལ་ལ་སྔ་བ་ཡར་ཀླུངས་སོག་ཀ་རུ། །མཁར་ལ་སྔ་བ་ཡམ་བུ་དགུ་བརྩེགས་བརྩིགས། །དེ་དང་གནམ་མུག་མུག་སྲས་མུ་ཁྲི་བཙན། །དེ་དང་ས་དིང་དིང་སྲས་དིང་ཁྲི་བཙན། །དེ་དང་སོ་ཐམ་ཐམ་སྲས་སོ་ཁྲི་བཙན། །དེ་དང་དོག་མེར་མེར་སྲས་མུར་ཁྲི་བཙན། །གདགས་ཀྱི་ལྷ་མོའི་སྲས་སུ་གདགས་ཁྲི་བཙན། །སྲིབས་ལྷ་སྔོན་མོའི་སྲས་སུ་སྲིབས་ཁྲི་བཙན། །རི་བྲག་མི་ཐོགས་མཁའ་ལ་གཤེགས་བཞུད་བྱེད། །ཉིན་བཞིན་སར་བྱོན་མཚན་བཞིན་མཁའ་ལ་གཤེགས། །སྲས་ཀྱི་ཆིབས་ཁ་ཐུབ་པར་གྱུར་བ་ན། །རྨུ་ཐག་ལ་འཇུ་ལྷ་ལ་གཤེགས་སོ་སྐད། །ལྷ་ལ་རོ་མེད་འཇའ་ཚོན་ལྟར་ཡལ་བས། །དེའི་ཕྱིར་གནམ་གྱི་ཁྲི་བདུན་ཞེས་སུ་གྲགས། །དེ་ཡི་བཙུན་མོ་ས་བཙུན་ཀླུ་རྗེའི་སྲས། །ཕྱི་མོས་མཚན་དུ་གྲི་གུམ་བཙན་པོར་གསོལ། །དེ་ཡི་བཙུན་མོ་ཀླུ་སྲིང་མེར་ལྕམ་ལ། །ཤ་ཁྲི་ཉ་ཁྲི་བྱ་ཁྲི་གསུམ་ཡང་བལྟམས། །མུ་སྟེགས་བོན་པོ་འ་ཞ་མཆོད་བཀུར་མཛད། །རྒྱལ་ཕྲན་ལོ་ངམ་འཐབ་པས་གྲི་རུ་བཀྲོངས། །ཟགས་ཀྱི་གའུར་སྤུར་བཅུག་ཉིང་ཆུར་བསྐུར། །བྱེ་མ་ལག་རིང་ཞེས་པས་མཛོད་དུ་ཉར། །ལྷ་སྲས་མཆོད་གསུམ་ཡབ་ཀྱི་དཀོར་མཛོད་ནས། །ལྷ་གླང་ཅེ་འུ་རུ་དཀར་ཞོན་ཏེ་བྲོས། །གཅེན་པོ་ཤ་ཁྲིས་ཀོང་གི་རྗེ་བོ་མཛད། །འབྲིང་བོ་བྱ་ཁྲིས་སྤོ་བོའི་མངའ་དབང་མཛད། །ཆུང་བ་ཤ་ཁྲིས་ཉང་བོ་རྗེ་མཛད་ཅིང་། །ལྷ་གཅིག་ཆུང་མར་ཡུམ་ལ་རྟ་རྫིར་བཅོལ། །རུ་ལས་སྐྱེས་ཀྱི་འཕྲུལ་དང་མཐུ་སྟོབས་ཀྱིས། །ལོ་ངམ་འཁོར་བཅས་འདུལ་ཞིང་ཚར་གཅོད་བྱས། །ཉང་རོ་ཤམ་པའི་མཁར་གྱི་རྩེ་ནས་ཕབ། །ལོ་ངམ་ཕོ་བརྒྱ་མོ་བརྒྱ་བསད་དོ་སྐད། །བྱ་ཁྲི་སྲིདབཞུགས་སྤུ་དེ་གུང་རྒྱལ་ཟེར། །རྒྱལ་བོའི་སྐུ་མཁར་འཆིང་བ་སྟག་རྩེ་བརྩིགས། །ཤམ་པོའི་མཁར་གྱི་རྩི་ཕབ་རྨང་རྡོ་བྱས། །རྒྱལ་བོ་མི་མཆོག་རུ་ལས་སྐྱེས་ཁྱོད་ཀྱིས། །ང་ཡི་ཡབ་ཀྱི་གོ་བགྱིས་རྒྱལ་སར་བཏོན། །སྲས་ཀྱི་གོ་བགྱིས་ཡབ་ཀྱི་སྐུ་འཚལ་གཉེར། །མིང་ཡང་ཁུ་བོར་བཀུར་བས་ཁུའི་རུས་ཐོགས། །ལྷ་མི་སྨོན་གཟུང་ཞེས་ཏེ་མཛངས་མིར་བསྔགས། །མཛངས་པའི་ལས་སུ་ཤིང་བསྲེགས་སོལ་བ་བྱུང་། །དེ་ཡིས་རྡོ་བཞུས་གསེརདངུལ་ཟངས་ལྕགས་ཕྱུང་། །ཤིང་ལ་མིག་ཕུག་གཤོལ་དང་གཉའ་ཤིང་བྱས། །ས་གཞི་བརྐོས་ནས་ཕུ་ཆུ་ཡུར་དུ་དྲངས། །རྨོན་པ་དོར་བསྡེབས་སྤང་ཐང་ཞིང་དུ་རྨོས། །བརྒལ་བས་མི་ཐར་ཞིང་ལ་ཟམ་པ་བཙུགས། །རྨོས་པའི་ལོ་ཏོག་ཐོག་མ་འདི་ནས་བྱུང་། །གནམ་བོན་གཤེན་པོ་ཆེ་དང་བོན་གཞུང་བྱུང་། །མ་སངས་མཆིལ་སྒྲུང་སོགས་ཀྱི་འཆད་ཉན་བྱུང་། །རིག་པའི་ཐབས་སུ་ལྡེ་འུ་ཡོངས་སུ་དར། །འདི་གཉིས་བང་སོ་རྫ་དང་གཡའ་ལ་བཏབ། །དེ་བས་སྟོབས་ཀྱི་སྟེངས་གཉིས་གྲགས། །དེ་དང་བཙུན་མོ་འོམ་ཐང་སྨན་མཚོའི་སྲས། །ལེགས་དྲུག་ཐོག་མ་ཨི་ཤོ་ལེགས་ཞེས་བྱ། །དེ་ཡི་བློན་པོ་ལྷ་བུ་མགོ་དཀར་ཞེས། །ཁུ་བོ་རུ་ལས་སྐྱེས་ཀྱི་བུ་ཡིས་བྱས། །འཕྲུལ་བློན་མཛངས་མི་བདུན་གྱི་གཉིས་པ་སྟེ། །མཛངས་པའི་ལས་སུ་ཞིང་ལ་དོར་ཁ་བརྩིས། །འབྲོག་ལ་ཐུལ་རྩིས་ཕུ་ཆུ་ཡུར་དུ་དྲངས། །མདའ་རུ་ཆུ་མ་རྨོ་བ་འདི་ནས་བྱུང་། །རྒྱལ་བོ་ཨི་ཤོ་ལེགས་དང་ཐི་ཤོ་ལེགས། །གུང་རི་ལེགས་དང་ལྔ་བ་འབྲོང་གཞིར་ལེགས། །ཐོ་ཤོ་ལེགས་ཏེ་དེ་རྣམས་བང་སོ་ནི། །སྤང་མཚམས་བཏབ་པས་ས་ཡི་ལེགས་དྲུག་སྟེ། །གནམ་སྨན་བུ་མོ་དག་ནི་ཡུམ་དུ་གྱུར། །ལེགས་དྲུག་དྲིན་ལ་ཕྱིང་བ་རྩེ་དྲུག་བྱུང་། །ཟ་ནམ་ཟིན་ལྡེ་འཕྲུལ་ནམ་གཞུང་བཙན་ལྡེ། །སེ་སྣོལ་ནམ་ལྡེ་སེ་སྣོལ་པོ་ལྡེ་སྟེ། །ལྡེ་ནོར་ནམ་དང་ལྡེ་སྣོལ་པོ་ཞེས་གྲགས་དང་། །ལྡེ་རྒྱལ་བོ་དང་ལྡེ་སྤྲིན་བཙན་དག་སྟེ། །བང་སོ་ཆུ་ཡི་གཞུང་ལ་བཏབ་པ་ལས། །མཚོ་ལ་ཁ་བ་བབས་པ་འདྲའོ་སྐད། །དེ་ཕྱིར་ཆུ་ལ་ལྡེ་བརྒྱད་ཅེས་སུ་གྲགས། །བཙན་ལྔའི་ཐོག་མ་ཏོ་རི་ལོང་བཙན་ཏེ། །འདིའི་ཡན་ཡུམ་རྣམས་ལྷ་མོ་ཀླུ་མོ་ཡིན། །དེའི་ཕྱིར་ལྷ་སྲས་ལྡེ་སྲས་ཞེས་སུ་གྲགས། །འབངས་དང་གཉེན་བསྲེས་མན་དུ་ལྡེ་སྲས་བཞག། །དེ་ཡི་དོད་དུ་ལྷ་སྲས་བཙན་པོ་ཟེར། །ཁྲི་བཙན་ནམ་དང་ཁྲི་སྒྲ་དཔུང་བཙན་ཏེ། །ཁྲི་ཐོག་རྗེ་བཙན་ཐོ་ཐོ་རི་གཉན་བཙན། །མཚན་གསོལ་ཚིགས་ལ་བཙན་ལྔ་ཞེས་སུ་གྲགས། །ཁྲི་ཐོག་རིང་ལ་རོལ་མོ་གླིང་བུ་བྱུང་། །ཐོ་ཐོའི་རིང་ལ་གཉན་པོ་གསང་བ་བྱུང་། །བཙན་ལྔའི་མན་ཆད་བང་སོ་ཕྱུང་ཡུལ་བཏབ། །མཆད་པ་ས་ཕུང་སྦྲ་འདྲ་ཡོད་ཅེས་གྲགས། །དེ་རྗེས་ཁྲིམས་མེད་རྒྱལ་བོ་གསུམ་བྱུང་སྟེ། །ཁྲི་གཉན་ང་བཟུང་བཙན་དེ་ཡི་ཐོག་མ་ཡིན། །དེ་ཡི་སྲས་ནི་འབྲོང་གཉན་བཟུང་བཙན་སྟེ། །རྡོབ་པའི་ནད་ཀྱིས་སྐུ་ལ་སྙུང་འཛེ་བྱུང་། །འགོས་པ་འགོག་ཆེད་གསོན་པོར་དུར་དུ་ཞུགས། །དེ་སྲས་དམུས་ལོང་དཀོན་པ་བཀྲ་བ་ཡིས། །གཉན་པོ་གསང་བ་མཆོད་པའི་བྱིན་རླབས་དང་། །འ་ཞའི་ཡུལ་གྱི་སྨན་པས་སྤྱན་ཕྱེས་ཏེ། །སྟག་མོའི་རི་ལ་རི་དྭགས་གཉན་གཟིགས་པས། །མཚན་ཡང་སྟག་རི་གཉན་གཟིགས་བཙན་དུ་ཐོགས། །རྒྱལ་ཕྲན་རྣམས་ཀྱི་སུམ་གཉིས་འོག་ཏུ་འདུས། །བན་པོ་རྗེ་འུ་ཞ་རྗེ་དྲང་གར་རྗེ། །ཟིན་པོ་རྗེ་དང་ཞང་ཞུང་རྗེ་རྣམས་བཏུལ། །ཉང་སྦས་གནོན་གསུམ་འབངས་སུ་ཆུད་དོ་སྐད། །འདི་དུས་ཁྲི་ངོར་སྣང་བཙན་མོང་གི་བུས། །བྲེ་ཕུལ་སྲང་བཅོས་འབྲུ་མར་ཚད་ཀྱིས་བཅལ། །དགའ་གཉིས་ཚོང་བྱས་མཐུན་གཉིས་ལ་ལུང་བསྡེབས། །དེ་གོང་བོད་ན་ཚོང་དང་བྲེ་སྲང་མེད། །དེའི་ཕྱིར་མཛངས་མི་བདུན་གྱི་གསུམ་པར་གྲགས། །དེ་དང་ཡུམ་སྟོང་བཙུན་འབྲོ་ལས་འཁྲུངས་པའི། །རྒྱལ་བོ་གནམ་རུའི་སྲོང་བཙན་སྐུ་རིང་ལ། །རྒྱ་ནག་ཡུལ་ནས་རྩིས་དང་སྨན་དཔྱད་བླངས། །དམག་དང་འཁྲུག་པ་གཡོ་བའི་འདུན་མ་བྱས། །བར་དུ་བསྐལ་བ་འཁོན་བྱུང་སྡེ་རིས་ལ། །མཐའ་རྒྱ་བཅད་ནས་རྒྱ་དང་གྲུ་གུ་བཏུལ། །ཕུང་གཡག་བཏུལ་ནས་ལན་ཚྭ་རྙེད་པ་ཡིན། །ཁྲི་བརྩིགས་འབུམ་གདུགས་བ་དམར་འོ་མས་བརྩིགས། །དེ་དང་འབྲི་བཟའ་ཐོད་དཀར་གཉིས་ཀྱིས་སྲས། །ཕོ་བྲང་མི་འགྱུར་བྱམས་པ་གླིང་དུ་འཁྲུངས། །ཁྲི་སྲོང་བཙན་ཞེས་མཚན་དུ་གསོལ་བ་ལས། །ཡོན་ཏན་ཀུན་ལྡན་ཤིན་ཏུ་ཐུགས་སྒམ་པས། །མཚན་ཡང་སྲོང་བཙན་སྒམ་པོ་ཞེས་སུ་ཆགས། །ཁེང་ལོག་འདུལ་ཏེ་རྒྱལ་ས་གཅིག་གྱུར་བྱས། །རྒྱལ་ས་ལྷ་སར་སྤོས་ནས་ཕོ་བྲང་བརྩིགས། །སྲིད་ཀྱི་སྒྲིག་གཞི་མ་ལུས་ཚང་བར་བྱས། །ཡི་གེ་སྲོལ་གཏོད་ཀུན་གྱིས་དེ་ལ་སྦྱངས། །མཐའ་ཡི་རིག་གནས་ཐབས་ཀྱིས་བོད་དུ་དྲངས། །ཚེ་སྟོད་རྒྱལ་ཁྲིམས་ཚེ་སྨད་ཆོས་ཁྲིམས་བཅས། །བོད་ཁམས་བཙན་ཞིང་སེམས་ཅན་བདེ་ལ་བཀོད། །ཐོག་མར་བོད་ཡུལ་རུ་ཆེན་ལྔ་རུ་ཕྱེས། །ཡུལ་གྱི་དབང་རིས་རྣམ་པ་བཅོ་བརྒྱད་བཅད། །རྒོད་ཀྱི་སྟོང་སྡེ་དྲུག་ཅུ་རྩ་གཅིག་ཕྱེ། །གཡུང་གི་མི་སྡེ་ཁེང་དང་ཡང་ཁེང་འབྱེད། །ཞང་གསུམ་བློན་བཅས་དབུས་ཀྱི་འདུན་ས་འཛིན། །དཔའ་སྡེ་གསུམ་གྱིས་ཕྱི་ཡི་སོ་ཁ་སྲུང་། །དེ་ལ་བོད་ཀྱི་ཁོས་དྲུག་ཅེས་སུ་གྲགས། །ཐོན་མིས་བོད་ཡིག་ཐོག་མ་མཛད་པའི་ཕྱིར། །བོད་མི་མཛངས་མི་བདུན་གྱི་བཞི་བར་གྲགས། །གཞན་ཡང་ཁྲི་བཟང་ཡང་སྟོན་གཉགས་ཀྱི་བུས། །མཛངས་པའི་ལས་སུ་རི་ཁྱིམ་ཀླུངས་སུ་ཕབ། །སྤོ་མཐོས་མཁར་དུ་བརྩིགས་ནས་གྲོང་ཁྱེར་བཅོས། །འདི་ཡན་བོད་ཀྱི་གྲོང་ཁྱེར་རི་ལ་ཡོད། །དེའི་ཕྱིར་བོད་ཀྱི་མཛངས་མི་ལྔ་བ་ཡིན། །གུང་རི་གུང་བཙན་རྒྱལ་སྲིད་དབང་བསྐུར་ཏེ། །དེ་ཡི་ལྷ་སྲས་མང་སྲོང་མང་བཙན་འཁྲུངས། །གུང་སྲོང་ཆབ་སྲིད་ལོ་ལྔ་མཛད་ནས་ནི། །བཅོ་བརྒྱད་བཞེས་ཚེ་ཕོ་ཏ་ལ་རུ་གཤེགས། །སླར་ཡང་ཡབ་ཀྱིས་རྒྱལ་སྲིད་ཆོས་བཞིན་བསྐྱངས། །མང་སྲོང་བཅུ་གསུམ་བཞེས་པས་རྒྱལ་སྲིད་བཟུང་། །ལོ་ནི་བཅོ་ལྔ་རྒྱལ་སྲིད་ཆོས་བཞིན་བསྐྱངས། །ཡབ་འདས་ཞག་བདུན་ལོན་པའི་ཉིན་དེ་ལ། །འདུས་སྲོང་མང་རྗེ་རླུང་ནམ་འཕྲུལ་གྱི་རྒྱལ། །སྦྲགས་ཕུར་འཁྲུངས་ཏེ་དེ་འཕྲལ་རྒྱལ་བོར་གསོལ། །དེའི་དུས་ལྷ་སྲིན་ངར་བསྡུས་གྱད་བདུན་བྱུང་། །ཁྲི་འབྲིང་སྤུན་གྱིས་བློན་པོར་ལོ་བཅུར་བྱས། །ལག་རིས་ཧོར་དང་ག་གོན་འདུལ་བར་བྱས། །བྱང་ཕྱོགས་རྒྱ་བསྐྱེད་ནང་ས་བདེ་བར་བསྐྱངས། །སྔ་གོང་བོད་རྒྱལ་བྱོན་ལས་མངའ་ཐང་ཆེ། །ཉེར་དགུ་སྲིད་བསྐྱངས་ལྗང་གི་ཡུལ་དུ་གཤེགས། །ལྷ་སར་མེས་ཀྱི་དགོངས་དུས་མཆོད་པ་བཙུགས། །མངའ་ཐང་ལྷ་འདྲ་མཐུ་རྩལ་ཕ་མཐའ་བྲལ། །མཐའ་ཡི་རྒྱལ་བོའི་དབང་དུ་འདུས་པར་གྱུར། །དེ་སྲས་ཁྲི་གཙུག་ལྡེ་བཙན་ཞེས་བྱ་བ། །ལྕགས་ཕོ་འབྲུག་ལ་ཕོ་བྲང་ལྡན་དཀར་འཁྲུངས། །ལྷ་ཁང་ལྔ་བཞེངས་མདོ་སྡེ་དུ་མ་དང་། །སྨན་དཔྱད་སོགས་ཕོ་བྲང་འཕང་ཕང་བརྩིགས། །རྒྱ་བཟའ་ཁབ་བསུས་གཤིན་པོའི་དགེ་སྲོལ་བཙུགས། །དེའི་སྲས་རྒྱལ་བོ་ཁྲི་སྲོང་ལྡེ་བཙན་གྱིས། །དབུ་རྨོག་བཙན་པོས་མཐའ་ཡི་དགྲ་བཞི་བཏུལ། །མཐའ་ཡི་རྒྱལ་བོ་རྣམས་ཀྱང་མངའ་རུ་བསྡུས། །དམ་པའི་ཆོས་རྣམས་དར་ཞིང་རྒྱས་པར་མཛད། །དགོན་གྱི་ཐོག་མ་དཔལ་གྱི་བསམ་ཡས་བཞེངས། །ལོ་པཎ་སྤྱན་དྲངས་མདོ་དང་གསང་སྔགས་བསྒྱུར། །རབ་བྱུང་ཐོག་མ་སད་མི་མི་བདུན་བྱུང་། །དམ་ཆོས་མི་བསྣུབས་བཀའ་གཙིགས་མནའ་ཡིས་བསྒྲགས། །རྒྱལ་ཁམས་དག་ནི་ས་ཚིགས་ཀྱིས་བཅད་ནས། །བོད་ཀྱི་ཁམས་ནི་གླིང་དགུར་ཕྱེ་བར་མཛད། །རྗེ་ཡི་གདུང་རབས་འབངས་ཀྱི་ས་བཅད་དང་། །ཡ་རབས་རྣམས་ལ་ཕྱག་དང་ཞེ་སའི་ཚུལ། །གཏམ་དང་ལོ་རྒྱུས་བཟང་བའི་ལས་ཐབས་དང་། །ཡ་རབས་བཟང་བོའི་ལུགས་ཀྱི་བསྟན་བཅོས་ལ། །བོད་འབངས་ཐམས་ཅད་བཀོད་དེ་བདེ་བར་མཛད། །མགོས་ཀྱི་བློན་པོ་ཁྲི་བཟང་ཡབ་ལྷག་གིས། །མཛངས་པའི་ལས་སུ་གསོས་སྟོང་སྟོང་ཐང་བཅད། །ཕྱི་རབས་བློན་པོ་གཡར་ཁྲལ་ཆུང་བར་བྱས། །སྟག་བཙན་གཉན་གཟིགས་བློན་པོ་གཉེར་གྱི་བུས། །མཛངས་པའི་ལས་སུ་དབྱར་རྩྭ་དགུན་དུ་བསྐམས། །མཛངས་མི་བདུན་ལས་ཐ་མ་གཉིས་སུ་གྲགས། །དེ་ནས་ལྷ་སྲས་མུ་ཏིག་བཙན་པོ་ཡིས། །ཞང་བློན་དབུ་རིང་བསད་པས་བྱང་དུ་སྤྱུགས། །འགོས་ཀྱིས་ཞལ་ལྕེ་དགའ་གསུམ་བྱ་བ་བཅད། །རྗེས་སུ་སྣ་ནམ་པ་ཡིས་བཀྲོངས་སོ་ཟེར། །སྤྲུལ་བའི་རྒྱལ་བོ་ཁྲི་སྲོང་ལྡེ་བཙན་དེས། །ལྷ་སྲས་ཐུ་བོ་མུ་ནེ་བཙན་པོ་ལ། །རྒྱལ་སྲིད་གཏད་དེ་ཆོས་ཀྱི་རྒྱལ་བོར་བསྐོས། །མུ་ནེ་བཙན་པོས་མཆོད་ཆེན་བཞི་བཙུགས་ཤིང་། །བོད་འབངས་ལྟོགས་ཕྱུག་ལན་གསུམ་བར་དུ་བསྙམས། །ལོ་གཅིག་ཟླ་དགུ་སྲིད་མཛད་ཡུམ་གྱིས་བཀྲོངས། །དེ་ནས་ཁྲི་ལྡེ་སྲོང་བཙན་རྒྱལ་སར་བསྐོས། །འཇིང་ཡོན་སད་ན་ལེགས་ཞེས་མཚན་དུ་གསོལ། །ལོ་པཎ་མང་དམ་ཆོས་དུ་མ་བསྒྱུར། །བསམ་ཡས་མཐིལ་དུ་དམ་ཆོས་གཞི་མ་བཙུགས། །གདོང་དམར་དམག་དཔུང་ཁྲི་ཚོ་བརྒྱ་ཆས་ཏེ། །ཕ་རོལ་དགྲ་འཇོམས་མཚོན་ཆས་འཁོར་ལོ་བསྐོར། །འཛམ་གླིང་སུམ་གཉིས་ཆབ་སྲིད་མངའ་རུ་བསྡུས། །ཆབ་སྲིད་མངའ་ཐང་དོ་ཟླ་མེད་པར་བྱས། །ཐང་ཁྲིམས་རྣམ་ལྔ་ནང་གི་བཅའ་ཁྲིམས་བཅས། །ཞལ་ལྕེ་རྣམ་ལྔས་ལེགས་ཉེས་འདུན་སྟངས་དབྱེ། །དབུས་ཀྱི་མངས་རིས་གསུམ་ནས་བསིངས་ཁུ་འདྲེན། །མཐའ་ཡི་མི་སྡེ་བཞི་ཡིས་དཀོར་ནོར་འབུལ། །མཚམས་ཀྱི་སོ་ཁ་དག་ལས་དཔྱ་ཡང་བསྡུས། །དཔའ་སྡེ་རྣམ་ལྔས་མཐའ་ཡི་སོ་ཁ་སྲུང་། །དཔའ་སྡེ་སྟག་སྡེ་མཐའ་རྒྱ་དམག་གིས་འདུལ། །ལྷ་ཆོས་མི་ཆོས་གཉིས་ཀ་དར་ཞིང་རྒྱས། །བོད་ཁམས་མ་ལུས་བདེ་སྐྱིད་ལོངས་སྤྱོད་ལྡན། །དེ་སྲས་ལྔ་ལས་གཉིས་ནི་དགུང་དུ་གཤེགས། །ལྷ་སྲས་གཙང་མས་རྒྱལ་སྲིད་རྩྭ་ལྟར་སྤངས། །དར་མ་སྡིག་ལ་དགའ་བས་རྗེར་མ་འོས། །བར་བ་ཁྲི་གཙུག་ལྡེ་བཙན་རལ་བ་ཅན། །བཅུ་གཉིས་ཡབ་འདས་དེ་དུས་རྒྱལ་སྲིད་བཞུགས། །བཙུན་མོ་ལྔ་བཞེས་དགེ་བཅུའི་བཀའ་ཁྲིམས་བཙན། །རྒྱ་ཡུལ་ཆམ་ལ་ཕབ་ནས་གཙིགས་བཟུང་སྟེ། །བོད་རྒྱའི་མཚམས་སུ་དབོན་ཞང་ལྷ་ཁང་བརྩིགས། །གནམ་ལ་ཆེ་བ་ཉི་ཟླ་ཟུང་གཅིག་སྟེ། །ས་ལ་ཆེ་བ་བཙན་པོ་དབོན་ཞང་གྲགས། །བཀས་བརྩིགས་ཡི་གེ་རྡོ་རིང་གསུམ་ལ་བརྐོས། །ཕྱིན་ཆད་བོད་རྒྱ་རང་རང་ཡུལ་ན་བདེ། །དམག་མི་བྱ་བར་གཙིགས་བཟུང་བྲོ་ཡིས་བསྩལ། །ལུགས་གཉིས་བཀའ་ཁྲིམས་ཆོས་བཞིན་གཏན་ལ་ཕབ། །མངའ་ཐང་དཔལ་འབྱོར་ས་ལ་བརྒྱ་བྱིན་བཞིན། །སྟོབས་དང་མཐུ་ཡིས་མཐའ་བཞི་ལས་རྒྱལ་ཏེ། །འཛམ་གླིང་སུམ་གཉིས་ལས་ལྷག་མངའ་དབང་མཛད། །དེ་ནས་འགྲོ་བའི་བསོད་ནམས་ཆབ་བཞིན་ཞི། །བློན་པོ་དུ་མས་ཆོས་ཁྲིམས་གཞིག་པར་བགྲོས། །ལོག་པའི་ཐབས་དང་སྙན་ཕྲ་གསོལ་བ་ཡིས། །ལྷ་སྲས་གཙང་མ་ཡུལ་ཕྱོགས་གཞན་དུ་སྤྱུགས། །བྲན་ཀ་དཔལ་ཡོན་ཉེས་མེད་སྐྱོན་གྱིས་བཀུམ། །རྒྱལ་བོའི་སྐུ་ལ་ནོངས་པ་བྱས་ནས་བཀྲོངས། །རྗེ་ཡི་མངའ་ཐང་དགུན་ཁའི་ཆུ་བཞིན་བྲི། །རྒྱལ་ཁྲིམས་སོག་རུལ་ཕོན་ཐག་བཞིན་དུ་ཆད། །དམ་ཆོས་བསྒྱུར་འཕྲོ་ཆད་ཅིང་ལོ་པ་བཞུད། །བོད་འབངས་བསོད་ནམས་མར་མེ་བཞིན་དུ་ཟད། །དར་མ་ཨུ་དུམ་བཙན་ལ་རྒྱལ་སྲིད་གཏད། །ལོ་གཉིས་འདས་ནས་དམ་ཆོས་འཇིག་པར་བརྩམས། །སྐུ་གསུང་ཐུགས་རྟེན་ལྷ་ཁང་མང་དུ་བཤིག། །གསང་རབ་ཕལ་ཆེར་མེར་བསྲེགས་ཆུ་རུ་བསྐུར། །ཆོས་པ་རབ་བསད་འབྲིང་སྤྱུགས་ཐ་མ་བཀོལ། །ར་ས་བསམ་ཡས་ཝ་ཚང་སྤྱང་ཚང་བཅས། །སྨར་གཡོ་གཙང་གསུམ་མདོ་ཁམས་སྨད་དུ་བྲོས། །དཔའ་རྡོར་བདུད་ཀྱིས་ཆས་ཞུགས་རྒྱལ་བོ་བཀུམ། །རྒྱལ་བོ་འདི་ལ་སྨད་ཀྱི་སྟེངས་ཞེས་ཟེར། །ཕྱི་ལོ་གདུང་བརྒྱུད་ཡུམ་བརྟན་འོད་སྲུངས་བྱུང་། །ཆེ་བར་མ་མཐུན་མངས་རིས་གཉིས་སུ་ཆད། །དབུ་གཡོ་གཉིས་སོགས་བོད་ཁམས་ཐམས་ཅད་རྩོད། །མཐའ་ཡི་སོ་ཤོར་འདུན་ས་ཆེ་ཆུང་ཞིག། །རྒྱལ་ཁྲིམས་ཆག་ཅིང་ཆོས་ཁྲིམས་དར་མདུད་གྲོལ། །མི་རྣམས་རྟག་ཏུ་ཁྲལ་དང་འུ་ལག་ལྕི། །རྗེ་ཁེང་ཁྱད་མེད་བོད་ཁམས་སིལ་བུར་ཐོར། །བང་སོ་རྣམས་ཀྱང་བགོས་ཏེ་ཕལ་ཆེར་བྲུས། །རྗེ་ཡི་མགུར་ལྷ་རྣམས་ཀྱི་གཡར་ངོ་ལས། །རྗེ་ཚན་རྣམ་དགུ་རྗེ་ལྷུས་བཅུ་རུ་ཆགས། །མངའ་བདག་འོས་སྲུངས་སྲས་པོ་དཔལ་མཁར་བཙན། །ལྷ་ཁང་འགའ་བཞེངས་སྟག་རྩེ་གཉགས་ཀྱིས་བཀྲོངས། །དེ་སྲས་མཆེད་གཉིས་སྟོད་ཀྱི་ཕྱོགས་སུ་བྱོན། །མངའ་རིས་སྐོར་གསུམ་ལ་སོགས་དབང་མཛད་དེ། །སྟོད་དུ་རྒྱལ་སྲིད་ཞྭ་ཙམ་ཆགས་ཞེས་གྲགས། །དེ་ཡི་གདུང་རྒྱུད་འགའ་ཞིག་སྨད་དུ་ཆགས། །མདོ་སྨད་ཙོང་ཁའི་ཡུལ་ཆེན་མངའ་མཛད་པས། །སྨད་དུ་རྒྱལ་སྲིད་ལྷམ་ཙམ་ཆགས་སོ་སྐད། །བར་དུ་དབུས་གཙང་རུ་བཞི་སིལ་བུར་གྱུར། །ཡུམ་བརྟན་འོད་སྲུངས་ལས་གྱེས་རྒྱལ་ཕྲན་ནི། །ཡུལ་གྲུ་རེ་ཡང་མང་དུ་གནས་པར་གྱུར། །སྨད་ཀྱི་སིལ་མར་གྲགས་པའི་དུས་ཚིགས་འདིར། །རང་སར་ཆེ་བའི་རྒྱལ་ཕྲན་མང་དུ་བྱུང་། །བོད་ཁམས་མི་མཐུན་སིལ་བུར་གནས་པའི་དུས། །གཅིག་ཏུ་མི་འདུས་ང་མཐོ་ང་བཙན་ཟེར། །གཞན་ལ་ཐུག་དུས་བཙན་པོ་གཅིག་ཀྱང་མེད། །ཆེན་པོ་ཧོར་དུས་ས་སྐྱས་དབང་དུ་བསྒྱུར། །འབྲི་གུང་རིན་སྤུངས་གཙང་བ་རྒྱལ་བོ་སྟེ། །དགའ་ལྡན་ཕོ་བྲང་ལ་སོགས་རིམ་བཞིན་དུ། །བོད་ཀྱི་ཁམས་ལ་བཀའ་ཡིས་དབང་དུ་བསྒྱུར། །བོད་ཁམས་མི་མཐུན་སིལ་བུར་གནས་པའི་དུས། །ཇིང་གིར་རྒྱལ་བོས་ཕྱོགས་ཀུན་ཆམ་ལ་ཕབ། །ཧོར་གྱིས་དམག་དཔུང་བོད་དུ་བཏང་བ་ལས། །འཐབ་པར་མ་ནུས་ཀུན་གྱིས་མགོ་བཏགས་ཞུས། །དེ་རྗེས་ས་ཕག་འབྲི་ཚལ་སྟག་གཡའ་བོས། །ཧོར་གྱི་རྒྱལ་སྲས་སོ་སོར་མགོ་བཏགས་བྱས། །གོ་ཏན་རྒྱལ་བོས་དོར་རྟ་ནག་པའི་དཔུང་། །མངགས་ནས་སུ་མཁས་རྩད་བཅད་ས་པཎ་མཆོག། །ཆེན་པོའི་ཧོར་གྱི་ཁབ་དུ་གདན་འདྲེན་བྱས། །ཆུ་ཡི་གླང་ལ་བོད་དུ་དམག་བཏང་ནས། །བོད་ཀྱི་རྒྱལ་ཁམས་གོང་བུ་གཅིག་ཏུ་བསྒྱུར། །སེ་ཆེན་ཁྲི་འཁོད་འགྲོ་མགོན་བླ་མས་བཀུར། །དབང་ཡོན་དང་བོར་ཁྲི་སྐོར་བཅུ་གསུམ་ཕུལ། །དབང་ཡོན་གཉིས་པར་ཆོལ་ཁ་གསུམ་པོ་ཕུལ། །གསུམ་པར་རྒྱ་ཡི་མི་ཡུར་རྒྱུན་བཅད་ཅིང་། །དེ་ནས་ས་སྐྱའི་བོད་དུ་དབང་དུ་བསྒྱུར། །དཔོན་རབས་དང་བོ་ཤཱཀྱ་བཟང་བོ་དང་། །ཀུན་དགའ་བཟང་བོ་ཞང་བཙུན་གསུམ་པ་ཡིན། །ཕྱུག་པོ་བན་དཀར་བྱང་རིན་ཀུན་གཞོན་ཏེ། །གཞོན་དབང་བྱང་རྡོར་དགུ་བ་ཨང་ལེན་ཡིན། །ལེགས་པ་དཔལ་དང་སེང་གེའི་དཔལ་དང་ནི། །ཡོན་བཙུན་འོད་ཟེར་སེང་གེ་རྒྱལ་བཟང་བ། །དབང་ཕྱུག་དཔལ་དང་བསོད་ནམས་དཔལ་བ་དང་། །རྒྱལ་བཟང་དབང་བརྩོན་ནམ་མཁའ་བརྟན་པ་བཅས། །དཔོན་རབས་ཉི་ཤུ་རྩ་གཅིག་བཀའ་ཡིས་བསྒྱུར། །འབྲི་གུང་ཀུན་རྡོར་རིན་ཆེན་ཞེས་བྱ་བས། །ཧུ་ལ་ཧུའི་དམག་ཁྲིར་ས་སྐྱར་དྲངས། །དེའི་ལན་ས་སྐྱའི་དཔོན་ཆེན་ཨང་ལེན་གྱིས། །ཐི་མུ་བྷོ་ཁས་དམག་དང་གཙང་དམག་གིས། །འབྲི་གུང་དམག་དྲངས་སྐྱ་སེར་ཁྲི་ལྷག་བསད། །འདི་ལ་འབྲི་གུང་གླིང་ལོག་ཅེས་སུ་གྲགས། །ནང་ཁུལ་འགལ་བ་ཉམས་པའི་རྒྱུ་ཡིན་ཏེ། །ས་སྐྱའི་བླ་བྲང་བཞི་བོ་གཉིས་སུ་གྱེས། །ཕག་གྲུའི་ཁྲི་དཔོན་བྱང་ཆུབ་རྒྱལ་མཚན་གྱིས། །གཡའ་བཟང་བཅོམ་ནས་རྒྱལ་བཟང་ཕྱོགས་སུ་ལངས། །ས་སྐྱའི་དཔོན་ཆེན་དབང་ཕྱུག་བརྩོན་འགྲུས་ཕྱོགས། །བཅོམ་ཏེ་སྲིད་དབང་ཕག་གྲུའི་ལག་ཏུ་ཤོར། །རྫོང་ཆེན་བཅུ་གསུམ་བརྩིགས་ཤིང་ལོ་གསུམ་རེར། །རྫོང་དཔོན་འགྲོ་བའི་ལམ་ལུགས་གཏན་ལ་ཕབ། །ཁྲིམས་ཡིག་ཞལ་ལྕེ་བཅོ་ལྔ་ལག་ཏུ་བསྟར། །རི་བོ་དགེ་ལྡན་པ་ཡི་གྲུབ་མཐའ་བྱུང་། །གོང་མ་དང་བོ་སི་ཏ་རྒྱལ་མཚན་ཡིན། །བསོད་ནམས་གྲགས་པ་གྲགས་པ་རྒྱལ་མཚན་ཏེ། །གྲགས་པ་འབྱུང་གནས་ཀུན་དགའ་ལེགས་པ་དང་། །ངག་གི་དབང་བོ་ངག་དབང་བཀྲ་ཤིས་གྲགས། །ཞབས་དྲུང་ངག་དབང་གྲགས་པ་བཅས་ཏེ་བཅུ། །ངག་གི་དབང་བོའི་སྐབས་སུ་གཙང་ཕྱོགས་ཀྱི། །སྲིད་དབང་ཅུང་ཉམས་དབུས་སུ་སྔར་བཞིན་གནས། །ངག་གྲགས་ཆུང་དུས་ཞྭ་དམར་ཆོས་གྲགས་དང་། །མཚོ་སྐྱེས་རྡོ་རྗེ་དེ་ཡི་ཚབ་དུ་བསྐོས། །དབུས་ཀྱིས་གཙང་གི་ཕྱོགས་སུ་དམག་དྲངས་ནས། །ལོ་ངོ་ཉི་ཤུར་སྨོན་ལམ་ཆེན་མོ་བཅད། །རིན་སྤུངས་བ་ཡིས་དབང་བསྒྱུར་གྲགས་ནའང་། །བོད་ཡོངས་མིན་པས་མཁས་པས་དོགས་པ་འཛིན། །ཡོངས་གྲགས་མཚོ་སྐྱེས་རྡོ་རྗེ་དོན་འགྲུབ་དང་། །ངག་དབང་རྣམ་རྒྱལ་དོན་འགྲུབ་ཚེ་བརྟན་རྡོར། །བཞི་བརྩི་ངག་དབང་སྐབས་སུ་འབྲི་གུང་དར། །མའི་གཅུང་ངག་འཇིག་དུས་སུ་མངའ་ཐང་ཤོར། །དེ་ལས་མངའ་ཐང་ཆེ་བ་དོན་རྡོར་ཉིད། །དེ་རྗེས་སྡེ་སྲིད་གཙང་བས་དབང་བསྒྱུར་ཏེ། །ཞིང་ཤག་ཚེ་བརྟན་རྡོ་རྗེས་དཔུང་ཆེན་གྱིས། །པ་སྣམ་རྫོང་བཅོམ་གཙང་གི་རྒྱལ་བོར་གྱུར། །དེའི་སྲས་ཕུན་ཚོགས་རྡོ་རྗེས་དམག་སྐུལ་ནས། །ཆུ་བྱི་འཕན་པོ་གཞིས་སྣེའུ་སོགས་བླངས། །ས་རྟ་དཔུང་ཆེན་སྐུལ་ནས་སེ་འབྲུ་བཅོམ། །བོད་ཁམས་དབང་བསྒྱུར་འདི་གཉིས་གང་རུང་བརྩི། །དེའི་སྲས་ཀརྨ་བསྟན་སྐྱོང་དབང་བོ་ཡིས། །བཅོ་ལྔའི་སྟེང་དུ་ཀླ་ཀློའི་ཞལ་ལྕེ་བསྣན། །གྲུབ་མཐའི་རིས་སུ་འཁྱེར་བས་ཀུན་ལ་འཚེ། །དེ་ཡི་རྐྱེན་ལས་དགའ་ལྡན་ཕོ་བྲང་བྱུང་། །སྐབས་དེར་ཆོག་ཐུ་མཚོ་སྔོན་པོར་འོངས་ཏེ། །མདོ་སྨད་ཕྱོགས་ཀྱི་བོད་སོག་དབང་དུ་བསྒྱུར། །ཏའོ་ལུགས་དད་པས་བོད་སོག་ཆོས་ལ་སྡང་། །མདོ་སྟོད་བེ་རུ་རྒྱལ་བོ་དོན་ཡོད་པ། །བོན་ལ་དད་ཅིང་ས་གར་རྙིང་དགེར་འཚེ། །དབུས་གཙང་དབང་བསྒྱུརགཙང་བ་སྡེ་སྲིད་ནི། །ཀར་ལུགས་དད་ཅིང་དགེ་ལུགས་པ་ལ་སྡང་། །དེ་གསུམ་མནའ་འབྲེལ་དབུས་གཙང་ཚོང་འགྲུལ་བཅད། །ཆོག་ཐུ་གཙང་བས་དགེ་ལུགས་བརླག་ལ་བརྩོན། །བེ་རི་གཙང་བ་གཏོར་ཏེ་དབུས་གཙང་དུ། །སེ་འབྲུ་དགེ་གསུམ་རྩ་མེད་གཏོང་ཞེས་སྒྲོག། ། དགེ་ལེགས་སྦྱིན་བདག་ཀོ་ཤྲཱི་བསྟན་འཛིན་གྱིས། །ཆོག་ཐུ་བེ་རི་གཙང་བ་རྒྱལ་བོ་བཅོམ། །མདོ་དབུས་ཁམས་གསུམ་ལྔ་བའི་ཕྱག་ཏུ་ཕལ། །བསོད་ནམས་ཆོས་འཕེལ་བོད་ཀྱི་སྡེ་སྲིད་བསྐོས། །དེ་ནས་གྲོང་སྨད་འཕྲིན་ལས་རྒྱལ་བོ་དང་། །བློ་བཟང་མཐུ་སྟོབས་བློ་བཟང་སྦྱིན་པ་དང་། །སངས་རྒྱས་རྒྱ་མཚོ་ངག་དབང་རིན་ཆེན་སྟེ། །སྡེ་སྲིད་སངས་རྒྱམ་ལྷ་བཟང་ཧན་གྱིས་བཀྲོངས། །ཚངས་དབྱངས་རྒྱ་མཚོ་རྒྱ་ནག་ཡུལ་དུ་ཁྲིད། །ལྷ་བཟང་སྲིད་དུས་ཇིང་གུར་བ་ཡིས་འཕྲོག །དེའི་ཚབ་སྟག་རྩེ་ལྷ་རྒྱལ་རབ་བརྟན་བསྐོས། །ཇིང་གིར་བ་ཡིས་རྙིང་མར་རྣམ་པར་འཚེས། །ཨ་མདོ་སོག་པོ་མན་ཇུའི་དམག་དཔུང་གིས། །རྒྱལ་བ་སྐལ་བཟང་རྒྱ་མཚོ་ལྷ་སར་ཞུས། །སྡེ་བ་སྟག་རྩེའི་སྲོག་སྐྱོབ་ངོ་ཆེན་ལ། །མ་གསན་མཚོ་སྔོན་བཀུར་གཟོ་ཆུང་བའི་ཕྱིར། །དཔོན་ཁག་རྣམས་ཀྱིས་མན་ཇུར་རྒོལ་བ་ལ། །གཡར་ཏམ་བཅས་པས་མཚོ་སྔོན་དུས་འཁྲུགས་བྱུང་། །རྗེས་སུ་མཚོ་སྔོན་ཕམ་པས་དགོན་པ་ཕལ། །བསྲེགས་ཤིང་སྐྱ་སེར་ཁྲི་ཕྲག་ཁ་ཤས་བསད། །དེ་ནས་མདོ་ཁམས་ཡུལ་སྨད་ཕལ་ཆེ་བ། །མན་ཇུའི་བཀའ་ཡི་འོག་ཏུ་སྡོད་པ་བྱུང། །སྡེ་སྲིད་ཁང་ཆེན་མན་ཇུ་བློ་དཀར་བས། །ང་ལུམ་སྦྱར་གསུམ་གྱིས་ནི་དམར་གསོད་བྱས་། །དེ་གསུམ་སྲིད་དུས་ཕོ་ལྟའི་འཕྲུལ་སྦྱོར་གྱིས། །མན་ཇུའི་དམག་སླེབས་དུས་གཅིག་སྲོག་ཐོག་བཏང་། །ཕོ་ལྷ་བསོད་ནམས་སྟོབས་རྒྱལ་སྡེ་སྲིད་བྱས། །བོད་སྡོད་ཨན་བན་ཐོག་མ་འདི་ནས་བྱུང་། །འདས་པས་དེའི་སྲས་འགྱུར་མེད་རྣམ་རྒྱལ་བསྐོས། །ལྷ་སའི་མན་ཇུའི་དམག་མི་ཕྱིར་འཐེན་དང་། །སོག་པོ་ཨ་རོད་འབྲེལ་ལམ་བཟོ་བ་དང་། །བོད་དམག་གསར་བསྐྲུན་ས་མཚམས་སྲུང་བ་སོགས། །ཐབས་ཇུས་བཀོད་པས་ཨན་བན་གཉིས་ཀྱིས་བསད། །དེ་བས་ལྷ་སའི་རྒྱ་མི་ཡོད་ཚད་བསད། །དེ་ནས་རིམ་བཞིན་རྒྱལ་བ་སྐལ་བཟང་དང་། །དེ་མོ་ངག་དབང་འཇམ་དཔལ་དགེ་ལེགས་དང་། །ཚེ་གླིང་ཁྲི་ཆེན་ངག་དབང་ཚུལ་ཁྲིམས་ཏེ། །དེ་ནས་དེ་མོ་འཇམ་དཔལ་རྒྱ་མཚོས་བསྐྱངས། །སླར་ཡང་ཚེ་གླིང་རྟ་ཚག་གཉིས་དེ་མོ་བ། །ངག་དབང་བློ་བཟང་ཚེ་གླིང་ངག་འཇམ་སྟེ། །པཎ་ཆེན་བསྟན་པའི་ཉི་མ་རྭ་སྒྲེང་བ། །རྒྱལ་བ་མཁས་གྲུབ་རྒྱ་མཚོས་འགན་བཞེས་ཤིང་། །དེ་ནས་ཡང་བསྐྱར་རྭ་སྒྲེང་སྲིད་ལ་བཞུགས། །ངག་དབང་ཡེ་ཤེས་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་དང་། །སྡེ་སྲིད་བཤད་སྒྲ་དབང་ཕྱུག་རྒྱལ་བོ་སྟེ། །སྡེ་སྲིད་སྡེ་དྲུག་མཁྱེན་རབ་དབང་ཕྱུག་དང་། །དེ་ནས་རྒྱལ་བ་འཕྲིན་ལས་རྒྱ་མཚོ་དང། །དེ་མོ་ངག་དབང་བློ་བཟང་འཕྲིན་ལས་ཏེ། །དེ་ནས་རྒྱལ་བ་ཐུབ་བསྟན་རྒྱ་མཚོ་སྟེ། །དབྱིན་ཇིས་དམག་བྱས་རྒྱལ་བ་སོག་པོར་ཕེབས། །རྒྱལ་ཚབ་དགའ་ཁྲི་བློ་བཟང་རྒྱལ་མཚན་བྱས། །དབྱིན་དམག་ལྷ་སར་དྲག་པོས་དབུས་འཇུག་བྱས་། །ལྷ་སའི་ཆིངས་ཡིག་དབང་མེད་འཇོག་དགོས་བྱུང། །རྒྱལ་ཚབ་དགའ་ཁྲི་ཚེ་སྨོན་གླིང་བས་བྱས། །བོད་སྡོད་ཨན་བན་དང་བཅས་ཕྱི་རུ་བསྒྲད་། །སིལ་ལའི་གྲོས་ཚོགས་ཕྱི་བོད་ནང་བོད་ཕྱེས། །རྒྱལ་བ་ཆེན་པོས་ལུགས་གཉིས་ཚུལ་བཞིན་བསྐྱངས་། །དེ་ནས་ར་སྒྲེང་སྟག་བྲག་རིམ་བཞིན་བྱོན། །ལམ་ལུགས་བཅོས་པར་ལུང་ཤར་དོན་རྐྱེན་བྱུང་། །བོད་ཚོགས་པ་ནུབ་ཕྱོགས་ཁག་དུ་སོང། །ལྷ་སར་རྒྱ་མི་སྐྲོད་པའི་དོན་རྐྱེན་བྱུང། །བཅིང་གྲོལ་དམག་གིས་ཆབ་མདོ་བཙན་གྱིས་བླངས། །ལྷ་མི་གུ་ཐུག་རྒྱལ་དབང་སྲིད་ལ་བསྐོས། །སྐུ་ཡི་ཚབ་དུ་སྲིད་བློན་རྣམ་གཉིས་བྱས། །པེ་ཅིན་མཁར་དུ་གྲོས་མཐུན་བཅུ་བདུན་བཀོད། །རྔ་ཁལ་སུམ་ཁྲིས་པཎ་ཆེན་བོད་དུ་སྐྱེལ། །བཅིང་བཀྲོལ་དམག་མི་བོད་ལྗོངས་སྲུང་བར་བསྐྱོད། །བོད་ཁམས་ཡོངས་སུ་རིམ་བཞིན་ཟིང་ཆ་ལངས། །དགའ་ལྡན་ཕོ་བྲང་གཞན་གྱི་ཡུལ་དུ་གྱར། །སྐར་ལྔའི་དར་དམར་པོ་ཏ་ལ་རུ་བཙུགས། །ལོ་རྒྱུས་ཁུངས་ལ་དེད་ན་རིན་ཐང་ཆེ། །དབྱེ་ཞིབ་རིགས་པས་དཔྱད་ན་བདེན་པ་རྙེད། །ལུང་རིགས་ཟུང་འབྲེལ་མེས་པོའི་ལུགས་སྲོལ་ཏེ། །གསར་བཤད་བཅོས་མའི་གཏམ་ལ་མགུ་བ་མིན། །དེ་སྐད་ལོ་རྒྱུས་ངག་གི་མེ་ལོང་འདི། །ལྷད་མེད་ཤེལ་དཀར་གཙང་མས་ཉེར་བཟོས་པ། །རང་གི་རིགས་ལ་ཞེན་པའི་རུས་པ་ཅན། །ན་གཞོན་བློ་ལྡན་དག་གི་ཆེད་དུ་བྱས། །
ཞེས་པ་ལོ་རྒྱུས་ངག་གི་མེ་ལོང་འདི་ནི་སློབ་བུ་དག་གི་ལོ་རྒྱུས་སྦྱོང་བའི་རྟེན་དུ་སྔོན་ནས་འབྲི་འདུན་གྱི་སྨོན་པ་ཡོད་ནའང་སྙོམས་ལས་སུ་ལུས་པ་ན་བོད་ཀྱི་རིག་གནས་ལ་ལྷག་བསམ་དཀར་བའི་རྫ་ཆུ་ཁ་བ། ཚོང་བ་ནམ་མཁའ་ཀློང་ཡངས་ཀྱིས་བསྐུལ་པའི་ངོར་ལོ་རྒྱུས་རྟོག་གེ་བ་ཤེས་རབ་བསྟན་དར་དུ་འབོད་པས་རང་སྤྱིལ་དཔེ་མང་སྤུངས་པའི་དགའ་ཚལ་དུ་བཀའ་ཐང་སྡེ་ལྔ་དང་ལྡེའུའི་ཆོས་འབྱུང་། མཁས་པའི་དགའ་སྟོན། བོད་ཀྱི་རྒྱལ་རབས་ལོ་རྒྱུས་མདོར་བསྡུས་གཞིར་བཟུང་ནས་ཤར་མར་བྲིས་སོ། །
