རྒྱ་གར་སྐད་དུ། མནྟ་སྱ་པ་ཐོ་པཱ་ཡ་བི་ཙིནྟ་ནཱ་མ། བོད་སྐད་དུ། སྔགས་ཀྱི་བཀླག་ཐབས་བསྡུས་པ་ཞེས་བྱ་བ། བླ་མ་དང་མགོན་པོ་འཇམ་པའི་དབྱངས་ ལ་ཕྱག་འཚལ་ལོ། །ཚོགས་གཉིས་སྤྲིན་དཀར་འཛུམ་པ་ལས་འོངས་མཚན་དཔེའི་རི་དྭགས་གཟུགས་ཀྱིས་བཀྲ། །ཁྱེད་སྐུ་ཟླ་བ་ཆ་ཤས་ཡོངས་རྫོགས་ཕྲིན་ ལས་འོད་དཀར་ཁོར་ཡུག་ཅན། །གང་གིས་མཐོང་ན་སྙིང་མུན་སེལ་བྱེད་གདུལ་བྱའི་མཁར་ཁྱབ་འགྲོས་གཉིས་པ། །དབྱངས་ཅན་ལྷ་མོ་མཚན་མོའི་ མགོན་ཁྱོད་དེ་རིང་བདག་བློའི་ཀུ་མུད་སྐྱོངས། །གསང་སྔགས་ཆོས་ཚུལ་ཟབ་མོའི་གཞལ་ཡས་ཁང་། །བདེ་བར་འཇུག་འདོད་གང་བློའི་བུ་མོ་ལ། །སྒྲ་ སྦྱོར་སྔགས་ཀྱི་བརྗོད་ཚུལ་མགུལ་རྒྱན་མཆོག །གནད་ཀྱི་ཕྲེང་བ་དེ་རིང་ཁོ་བོས་སྦྱིན། །དེ་ཡང་འདི་ལྟར་ཡི་གེའི་ངོ་བོ་ནི་གང་། དེ་ཁམས་གང་གིས་ བསྡུས་ཤིང་སྐྱེ་བའི་ཚུལ་ནི་གང་། བཀླག་པའི་ཐབས་ནི་ཅི་ཞེ་ན། ངོ་བོ་ཨཱ་ལི་ཀཱ་ལི་སྟེ། །ཕོ་མོ་མ་ནིང་གསུམ་དུ་གནས། །ཨ་ཀུ་ཧ་ྈྐ་ནམ་མཁའི་ཁམས། ། མགྲིན་པའི་གནས་ལས་སྐྱེ་བ་ཡིན། །ཨི་ཙུ་ཡ་ཤ་རླུང་གི་ཁམས། །ལྕེ་རྩེ་སོ་རྐན་དག་ལས་སྐྱེ། །རྀ་ཊུ་ར་ཥ་མེ་ཡི་ཁམས། །སྤྱི་བོ་དག་ལས་སྐྱེ་བ་ཡིན། །ཨུ་པུ་ ཝ་ྉྤ་ཆུ་ཡི་ཁམས། །མཆུ་ཡི་གནས་ལས་སྐྱེ་བ་ཡིན། །ལྀ་ཏུ་ལ་ས་ས་ཡི་ཁམས། །ལྕེ་རྩེ་སོ་ལས་སྐྱེ་བ་ཡིན། །འབྱུང་བ་ཆུང་བ་སྡེ་ལྔ་ནི། །ཀ་ཙ་ཊ་ཏ་པ་ རྣམས་ས། །ཁ་སོགས་ཆུ་ཡིན་ག་སོགས་མེ། །གྷ་སོགས་རླུང་ཡིན་ང་སོགས་མཁའ། ། ༈ ཨོཾ་དང་ན་མོ་ཏདྱ་ཐཱ། སྤངས་པའི་སྔགས་ཀྱི་ཐོག་མ་དང་། །རང་གི་ མིང་ཐོག་གཟུང་བར་བྱ། །གསལ་བྱེད་བརྩེགས་སམ་འཕུལ་བཞག་ཡོད། །དང་པོའི་ཡི་གེ་གཟུང་བར་བྱ། །གསལ་བྱེད་དགྲར་གྱུར་ནད་མང་ཞིང་། ། དབྱངས་ཡིག་དགྲར་གྱུར་སྲོག་དང་འབྲལ། །ཐ་མལ་གྱུར་ན་འབྲས་བུ་མེད། །གྲོགས་སུ་གྱུར་ན་དངོས་གྲུབ་འབྱུང་། །དེ་ཕྱིར་དགྲ་གྲོགས་བརྩི་བ་གཅེས། ། ས་དགྲ་མེ་ཡིན་མེ་དགྲ་ཆུ། །ཆུ་དགྲ་རླུང་ཡིན་རླུང་དགྲ་ས། །ས་ནི་མེ་ལ་ནུས་པ་མེད། །མེ་ནི་ཆུ་ལ་ནུས་པ་མེད། །ཆུ་ནི་རླུང་ལ་ནུས་པ་མེད། །རླུང་ནི་ས་ལ་ ནུས་པ་མེད། །ས་ཆུ་ཕན་ཚུན་གྲོགས་པོ་སྟེ། །མེ་རླུང་ཕན་ཚུན་གྲོགས་པོ་ཡིན། །ནམ་མཁའ་ཀུན་གྱི་གྲོགས་ཡིན་ཏེ། །ཀུན་ཀྱང་ནམ་མཁའི་གྲོགས་སུ་ འགྱུར། །དེ་ལྟར་དགྲ་དང་གྲོགས་ཤེས་པ། །སྔགས་ཀྱི་ཀློག་ཚུལ་འདི་ལྟ་སྟེ། །ཡི་གེ་ཐུང་བ་གཅིག་ཡུན་བཀླག །རིང་བ་གཉིས་ཀྱི་ཡུན་དུ་བཀླག །ཤིན་ཏུ་ རིང་བ་གསུམ་ཡུན་བཀླག །པླུ་ཏ་ཡང་དེ་བཞིན་ནོ། ། ༈ །འབྲེང་བུ་ན་རོ་ཡོད་གྱུར་ན། །རིང་པོ་ཉིད་ཕྱིར་འ་འདོགས་འཁྲུལ། །མིང་གི་ཐོག་མར་སྡེ་པ་ ཡི། །གསུམ་པ་འས་ཕུལ་བཞིན་དུ་བཀླག །དབུས་མཐར་བྱུང་ན་བོད་དང་མཚུངས། །བཞི་པ་རྣམས་ནི་གསུམ་པ་ལ། །ར་མགོ་ཧ་སྒྲ་ལྡན་པར་བཀླག །ཊ་ སོགས་ལོགས་སྡེའི་བཀླག་ཚུལ་ནི། །ལྕེ་རྩེ་རྐན་སྦྱར་སྟིབས་ཀྱིས་བསྒྱུར། །ཥ་ནི་ཁ་དང་ཆ་མཐུན་པར། །རྐན་གྱི་ཕུགས་ནས་སྟོངས་ཀྱིས་འདོན། །ཀྵ་ནི་ཁྱ་ དང་ཆ་མཐུན་ཙམ། །ཛ་ལ་ཉ་བཏགས་འགྱ་འམ་གཉའ། །གསལ་བྱེད་མི་འདྲ་དུ་མ་བརྩེགས། །རང་སྒྲ་ཐོན་ཙམ་མྱུར་བར་བཀླག །དབྱངས་དང་ང་རོ་ འོག་མ་དབང་། །འདྲ་བའི་གསལ་བྱེད་བརྩེགས་གྱུར་ན། །ཁ་ཁེངས་མཐོན་པོའི་གདངས་ཀྱིས་བཀླག །བརྩེགས་པའི་ཐོག་མ་གོང་མ་དབང་། །དེ་འོག་ རྣམས་ནི་དཀྱུས་སུ་བཀླག །ཉ་ཎ་ཡར་སྦྱར་ན་ལྟར་བཀླག །ཤ་ཥ་ཡར་སྦྱར་ས་ལྟར་བཀླག །ཀ་ནི་ག་དང་ཙ་ཛ་ཊ། །ཌ་ཏ་རྣམས་ནི་ད་རུ་བཀླག །པ་ཡིག་ རྣམས་ནི་བ་རུ་བཀླག །གྷ་ཛྷ་ཌྷ་དྷ་བྷ་རྣམས་ནི། །ཡི་གེ་གཉིས་གཉིས་འདོད་པ་འཁྲུལ། །དེ་བཞིན་དབྱངས་ཀྱི་རྀ་ལྀ་གཉིས། །ར་དང་ལ་རུ་འདོད་པ་ འཁྲུལ། །རྣམ་བཅད་ས་རུ་བཀླག་པའམ། །དྲག་པོར་བཅད་དེ་བཀླག་པར་བྱ། །རྗེས་སུ་ང་རོ་སྡེ་པ་ལ། །རང་རང་སྡེ་པའི་ལྔ་པར་བཀླག །ཡ་སོགས་བརྒྱད་ ལ་ང་རུ་བཀླག །ཡང་ན་སྡེ་ལྔའི་ཟླ་ལ་སྦྱར། །ཤད་ཀྱིས་ཆོད་ན་རང་ཉིད་བརྩི། །ཤད་མེད་ན་ནི་འོག་མ་བརྩི། །ཨོཾ་དང་ཧཱུཾ་གཉིས་མ་དང་ང་། །ཚིག་གྲོགས་ དག་ལ་བལྟོས་པ་མིན། །ལྕེ་རྩ་ཅན་དང་མཆུ་ཅན་གཉིས། །ཨ་ཀ་ཨ་པ་ཞེས་པའམ། །ཀོ་ཝ་པོ་ཝ་ཞེས་པར་བཀླག །ཡ་བཏགས་ཕལ་ཆེར་འགྲེལ་བུ་ནི། ། ཡའི་སྒྲ་ཅུང་ཟད་ལྡན་པར་བཀླག །ཡ་ལ་ཡ་བཏགས་ཡ་རང་ཉིད། །ཀ་ཁའི་ཡ་བཏགས་བོད་བཞིན་བཀླག །ར་ལ་ཡ་བཏགས་ར་ཡར་བཀླག །ཝ་བཏགས་ ཕལ་ཆེར་ན་རོ་སྟེ། །ཝ་སྒྲ་ཅུང་ཞིག་ལྡན་པར་བཀླག །ཀ་སོགས་སྲོག་ཏུ་རྀ་ལྀ་ཞུགས། །གི་གུ་ཕྱིར་ནི་ལོག་གྱུར་ན། །ཀི་རྀ་ཀི་ལྀ་ཞེས་མྱུར་བཀླག །ཀ་སོགས་ ར་དང་ལ་བཏགས་ནས། །གི་གུ་ཕྱི་ནི་མ་ལོག་ན། །ཀ་རི་ཀ་ལི་ཞེས་པར་བཀླག །ཨཻ་དང་ཨཽ་ཡི་བཀླག་ཚུལ་ནི། །ཨེ་ཨི་ཨོ་ཨུ་ཞེས་པར་བཀླག །དེ་བཞིན་ཀ་ སོགས་སྲོག་ཞུགས་ན། །ཀེ་ཨི་ཀོ་ཨུ་སོགས་སུ་བཀླག །རྒྱས་པར་སློབ་དཔོན་ངག་ལས་སློབས། །མདོར་བསྡུས་ཙམ་འདི་སྐད་གཉིས་པ། །སྣར་ཐང་ལོ་ཙཱ་ གང་དེ་ལ། །བཤེས་གཉེས་བློ་གྲོས་དཔལ་འབར་གྱིས། །ཐང་པོ་ཆེ་རུ་བསྐུལ་ནས་སྦྱར། །དེ་ལས་བྱུང་བའི་དགེ་བ་ཡིས། །འགྲོ་རྣམས་སྔགས་དོན་རྣམས་ གཟིགས་ཤོག །སྔགས་ཀྱི་ཀློག་ཐབས་ཞེས་བྱ་བ་རྫོགས་སོ།། །།
